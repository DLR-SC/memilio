${includes}

namespace py = pybind11;

${pretty_name}

PYBIND11_MAKE_OPAQUE(std::vector<mio::Graph<mio::SimulationNode<${simulation_class}>, mio::MigrationEdge>>);

PYBIND11_MODULE(_simulation_${pymio_name}, m)
{
    ${enum_population_states}

    ${indexing}

    pymio::bind_ParameterSet<${namespace}Parameters>(m, "Parameters");

    pymio::bind_Population${population_template_arguments}(m, "Population");

    using Populations = mio::Populations${population_template_arguments};
    pymio::bind_CompartmentalModel<Populations, ${namespace}Parameters>(m, "ModelBase");
    py::class_<${model_class}, mio::CompartmentalModel<Populations, ${namespace}Parameters>>(m, "${model_cpp_name}")
    ${model_init}

    pymio::bind_Simulation<${simulation_class}>(m, "Simulation");

    m.def(
        "simulate",
        [](double t0, double tmax, double dt, const ${model_class}& model) {
            return mio::simulate(t0, tmax, dt, model);
        },
        "Simulates a ${model_cpp_name} from t0 to tmax.", py::arg("t0"), py::arg("tmax"), py::arg("dt"), py::arg("model"));

    pymio::bind_ModelNode<${model_class}>(m, "ModelNode");
    pymio::bind_SimulationNode<${simulation_class}>(m, "SimulationNode");
    pymio::bind_ModelGraph<${model_class}>(m, "ModelGraph");
    pymio::bind_MigrationGraph<${simulation_class}>(m, "MigrationGraph");
    pymio::bind_GraphSimulation<mio::Graph<mio::SimulationNode<${simulation_class}>, mio::MigrationEdge>>(m, "MigrationSimulation");

    m.attr("__version__") = "dev";
}
