35def get_jh_data(read_data=dd.defaultDict['read_data'], 

36 file_format=dd.defaultDict['file_format'], 

37 out_folder=dd.defaultDict['out_folder'], 

38 no_raw=dd.defaultDict['no_raw'], 

39 start_date=date(2020,1,22), 

40 end_date=dd.defaultDict['end_date'], 

41 impute_dates=dd.defaultDict['impute_dates'], 

42 moving_average=dd.defaultDict['moving_average'], 

43 make_plot=dd.defaultDict['make_plot']): 

44 """! Download data from John Hopkins University 

45 

46 Data is either downloaded and afterwards stored or loaded from a stored filed. 

47 The file is "FullData_JohnHopkins.json" 

48 

49 Working with the data includes 

50 - rename columns such that "/" is deleted, e.g Country/Region becomes CountryRegion 

51 - data of all countries together are written to a file 

52 - download the data from following countries in a separate file 

53 and are stored in the according folders with the country name 

54 - Germany, SouthKorea, Spain, France, Italy, US, China 

55 - furthermore, all countries, for which provinces are added, are written to a file 

56 

57 @param read_data True or False. Defines if data is read from file or downloaded. Default defined in defaultDict. 

58 @param file_format File format which is used for writing the data. Default defined in defaultDict. 

59 @param out_folder Folder where data is written to. Default defined in defaultDict. 

60 @param no_raw True or False. Defines if unchanged raw data is saved or not. Default defined in defaultDict. 

61 @param start_date Date of first date in dataframe. Default defined in defaultDict. 

62 @param end_date Date of last date in dataframe. Default defined in defaultDict. 

63 @param impute_dates [Currently not used] True or False. Defines if values for dates without new information are imputed. Default defined in defaultDict. 

64 @param moving_average [Currently not used] Integers >=0. Applies an 'moving_average'-days moving average on all time series 

65 to smooth out effects of irregular reporting. Default defined in defaultDict. 

66 @param make_plot [Currently not used] True or False. Defines if plots are generated with matplotlib. Default defined in defaultDict. 

67 """ 

68 

69 filename = "FullData_JohnHopkins" 

70 

71 if read_data: 

72 file_in = os.path.join(out_folder, filename + ".json") 

73 # if once dowloaded just read json file 

74 try: 

75 df = pandas.read_json(file_in) 

76 # pandas>1.5 raise FileNotFoundError instead of ValueError 

77 except (ValueError, FileNotFoundError) as err: 

78 raise FileNotFoundError("Error: The file: " + file_in + \ 

79 " does not exist." + \ 

80 " Call program without -r flag to get it.") \ 

81 from err 

82 else: 

83 # Get data: 

84 # https://raw.githubusercontent.com/datasets/covid-19/master/data/time-series-19-covid-combined.csv 

85 df = gd.loadCsv('time-series-19-covid-combined', 

86 apiUrl='https://raw.githubusercontent.com/datasets/covid-19/master/data/') 

87 

88 # convert "Datenstand" to real date: 

89 # df.Datenstand = pandas.to_datetime( df.Datenstand, format='%d.%m.%Y, %H:%M Uhr').dt.tz_localize 

90 # ('Europe/Berlin') 

91 

92 # output data to not always download it 

93 if not no_raw: 

94 gd.write_dataframe(df, out_folder, filename, "json") 

95 

96 df.rename({'Country/Region': 'CountryRegion', 'Province/State': 'ProvinceState'}, axis=1, inplace=True) 

97 print("Available columns:", df.columns) 

98 

99 # extract subframe of dates 

100 df=mdfs.extract_subframe_based_on_dates(df, start_date, end_date) 

101 

102 

103 # Change "Korea, South" to SouthKorea 

104 df.loc[df['CountryRegion'] == "Korea, South", ['CountryRegion']] = 'SouthKorea' 

105 

106 # Generate folders if needed 

107 directory_ger = os.path.join(out_folder, 'Germany/') 

108 directory_es = os.path.join(out_folder, 'Spain/') 

109 directory_fr = os.path.join(out_folder, 'France/') 

110 directory_it = os.path.join(out_folder, 'Italy/') 

111 directory_us = os.path.join(out_folder, 'US/') 

112 directory_rok = os.path.join(out_folder, 'SouthKorea/') 

113 directory_prc = os.path.join(out_folder, 'China/') 

114 

115 # dictionary of countries 

116 countries = { 

117 "Germany": directory_ger, 

118 "Spain": directory_es, 

119 "France": directory_fr, 

120 "Italy": directory_it, 

121 "US": directory_us, 

122 "SouthKorea": directory_rok, 

123 "China": directory_prc, 

124 } 

125 

126 ########### Countries ########################## 

127 

128 gb = df.groupby(['CountryRegion', 'Date']).agg({"Confirmed": sum, "Recovered": sum, "Deaths": sum}) 

129 

130 gd.write_dataframe(gb.reset_index(), out_folder, "all_countries_jh", file_format) 

131 

132 for key in countries: 

133 # get data for specific countries 

134 gb_country = gb.reset_index()[gb.reset_index()["CountryRegion"] == key] 

135 dir_country = countries[key] 

136 gd.check_dir(dir_country) 

137 gd.write_dataframe(gb_country, dir_country, "whole_country_" + key + "_jh", file_format) 

138 

139 # Check what about external provinces. Should they be added? 

140 

141 ################ Countries with given States ################################ 

142 

143 # Get all countries where States are given 

144 dfD = df[~df["ProvinceState"].isnull()] 

145 

146 gb = dfD.groupby(['CountryRegion', 'ProvinceState', 'Date']).agg( 

147 {"Confirmed": sum, "Recovered": sum, "Deaths": sum}) 

148 

149 gd.write_dataframe(gb.reset_index(), out_folder, "all_provincestate_jh", file_format) 

150

152 # print(gb.reset_index()[gb.reset_index().ProvinceState=="Saskatchewan"])
