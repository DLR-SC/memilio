56def create_intervals_mapping(from_lower_bounds, to_lower_bounds): 

57 """! Creates a mapping from given intervals to new desired intervals 

58 

59 @param from_lower_bounds lower bounds of original intervals 

60 @param to_lower_bounds desired lower bounds of new intervals 

61 @return mapping from intervals to intervals 

62 The mapping is given as a list of tupels for every original interval. 

63 The list contains a tuple for every new interval intersecting the  

64 original interval. Each tuple defines the share of the original interval 

65 that is mapped to the new interval and the index of the new interval. We 

66 assume that the range of the intervals mapped from is contained in the  

67 range of the intervals mapped to. 

68 For example for from_lower_bounds = [5,20,30,80,85,90] and  

69 to_lower_bounds=[0,15,20,60,100] given the mapping would be 

70 [[[2/3,0], [1/3,1]], 

71 [[1,2]], 

72 [[3/5,2], [2/5,3]], 

73 [[1,3]], 

74 [[1,3]]] 

75 """ 

76 if (from_lower_bounds[0] < to_lower_bounds[0] or 

77 from_lower_bounds[-1] > to_lower_bounds[-1]): 

78 raise ValueError("Range of intervals mapped from exeeds range of" + 

79 "intervals mapped to.") 

80 # compute the shares of the original intervals mapped to the new intervals 

81 from_to_mapping = [[] for i in range(0, len(from_lower_bounds)-1)] 

82 to_idx = 0 # iterator over new intervals 

83 # iterate over original intervals 

84 for from_idx in range(0, len(from_lower_bounds) - 1): 

85 remaining_share = 1 # share of original interval to be distributed 

86 # position up to which the distribution is already computed 

87 pos = from_lower_bounds[from_idx] 

88 len_orig_interval = from_lower_bounds[from_idx+1] - pos 

89 # find first new interval intersecting the original interval 

90 while pos >= to_lower_bounds[to_idx+1]: 

91 to_idx += 1 

92 while from_lower_bounds[from_idx+1] > to_lower_bounds[to_idx+1]: 

93 # compute share of original interval that is send to new interval 

94 share = (to_lower_bounds[to_idx+1] - pos) / len_orig_interval 

95 from_to_mapping[from_idx].append([share, to_idx]) 

96 remaining_share -= share 

97 pos = to_lower_bounds[to_idx+1] 

98 to_idx += 1 

99 # if upper bound of the new interval is not smaller than upper bound of 

100 # the original interval assign remaining share of the original interval 

101 # to the new interval 

102 from_to_mapping[from_idx].append([remaining_share, to_idx]) 

103 return from_to_mapping

106def compute_vaccination_ratios( 

107 age_group_list, vaccinations_table, vacc_column, region_column, 

108 population, merge_2022=False): 

109 """! Computes vaccination ratios based on the number of vaccinations  

110 and the corresponding population data 

111 

112 @param age_group_list List of age groups considered. 

113 @param vaccinations_table Table of vaccinations (possible multiple columns 

114 for different number of doses) 

115 @param vacc_column Column name of vaccinations_table to be considered. 

116 @param region_column Column of regions in vaccinations table, e.g., ID_County or ID_State. 

117 @param population Table of population data for the given regions and considered age groups. 

118 @param merge_2022 [Default: False] Defines whether population data has to be merged to counties as of 2022. 

119 @return All vaccination ratios per region and age group. 

120 """ 

121 # create new data frame and reshape it 

122 df_vacc_ratios = pd.DataFrame(columns=[region_column] + age_group_list) 

123 df_vacc_ratios[region_column] = vaccinations_table[region_column].unique() 

124 df_vacc_ratios[age_group_list] = np.array( 

125 vaccinations_table[vacc_column]).reshape( 

126 len(vaccinations_table[region_column].unique()), 

127 len(age_group_list)) 

128 # compute county and age-group-specific vaccination ratios 

129 if merge_2022: 

130 population = geoger.merge_df_counties_all( 

131 population, sorting=[region_column], 

132 columns=region_column) 

133 df_vacc_ratios[['r'+age for age in age_group_list] 

134 ] = df_vacc_ratios[age_group_list] / population[age_group_list].values 

135 

136 return df_vacc_ratios

141 """! Splits a dataframe into separate dataframes. For each unique value that appears in a selected column, the corresponding value in another column is transfered to a new dataframe.

145 @param column_vals_name The name of the original column which will be renamed in the separate dataframes according to new_column_labels. 

146 @param new_column_labels new labels for resulting columns. There have to be the same amount of names and unique values. 

147 @return an array with the new splitted dataframes 

148 """ 

149 # TODO: Maybe we should input a map e.g. 1->Vacc_partially, 2->Vacc_completed etc. This is more future proof.

151 column_identifiers = df_to_split[column_to_split].unique() 

152 if len(column_identifiers) != len(new_column_labels): 

153 print("Wrong number of new labels handed over. Naming new columns 'new_column_i' for i=1,2,...") 

154 new_column_labels = [] 

155 for i in column_identifiers: 

156 new_column_labels.append('new_column_'+str(i))

158 df_subset = [] 

159 for i in range(0, len(column_identifiers)): 

160 df_subset.append(

165 columns={column_vals_name: new_column_labels[i]})
