80 """! download one dataset 

81 

82 Data is either downloaded from website or loaded as specific json file from the given "directory". 

83 The final data is also stored in "directory". 

84 After load or download the columns are renamed. 

85 

86 @param read_data False [Default] or True. Defines if data is read from file or downloaded. 

87 @param file_format File format which is used for writing the data. Default defined in defaultDict.

126def get_new_counties(data): 

127 """! Creates 7 new counties that were formed since 2011 and deletes old counties 

128 

129 Downloaded data is from 2011. 

130 However, new counties have been defined. 

131 Thus, the new ones have to be added, the data has to be calculated and the old counties have to be deleted. 

132 

133 @param data_temp Pandas dataframe 

134 @return Changed data 

135 """ 

136 

137 # create 7 new counties 

138 data_temp = np.append(data, np.zeros((7, data.shape[1])), axis=0) 

139 

140 # Göttingen 

141 data_temp[-7, 0] = 3159 

142 

143 # Mecklenburgische Seenplatte 

144 data_temp[-6, 0] = 13071 

145 

146 # Landkreis Rostock 

147 data_temp[-5, 0] = 13072 

148 

149 # Vorpommern Rügen 

150 data_temp[-4, 0] = 13073 

151 

152 # Nordwestmecklenburg 

153 data_temp[-3, 0] = 13074 

154 

155 # Vorpommern Greifswald 

156 data_temp[-2, 0] = 13075 

157 

158 # Ludwigslust-Parchim 

159 data_temp[-1, 0] = 13076 

160 

161 to_delete = [] 

162 

163 for i in range(len(data_temp[:, 0])): 

164 # fuse "Göttingen" and "Osterode am Harz" into Göttingen 

165 if data_temp[i, 0] in [3152, 3156]: 

166 data_temp[-7, 1:] += data_temp[i, 1:] 

167 to_delete.append(i) 

168 

169 # fuse "Müritz", "Neubrandenburg", "Mecklenburg-Sterlitz" 

170 # and "Demmin" into "Mecklenburgische Seenplatte" 

171 if data_temp[i, 0] in [13056, 13002, 13055, 13052]: 

172 data_temp[-6, 1:] += data_temp[i, 1:] 

173 to_delete.append(i) 

174 

175 # fuse "Bad Doberan and Güstrow" into "Landkreis Rostosck" 

176 if data_temp[i, 0] in [13051, 13053]: 

177 data_temp[-5, 1:] += data_temp[i, 1:] 

178 to_delete.append(i) 

179 

180 # fuse "Rügen", "Stralsund" and "Nordvorpommern" into 

181 # "Vorpommern Rügen" 

182 if data_temp[i, 0] in [13061, 13005, 13057]: 

183 data_temp[-4, 1:] += data_temp[i, 1:] 

184 to_delete.append(i) 

185 

186 # fuse "Wismar" and "Nordwestmecklenburg" into 

187 # "Nordwestmecklenburg" 

188 if data_temp[i, 0] in [13006, 13058]: 

189 data_temp[-3, 1:] += data_temp[i, 1:] 

190 to_delete.append(i) 

191 

192 # fuse "Ostvorpommern", "Uecker-Randow" and "Greifswald" 

193 # into "Vorpommern Greifswald" 

194 if data_temp[i, 0] in [13059, 13062, 13001]: 

195 data_temp[-2, 1:] += data_temp[i, 1:] 

196 to_delete.append(i) 

197 

198 # fuse "Ludwigslust" and "Parchim" into "Ludwigslust-Parchim" 

199 if data_temp[i, 0] in [13054, 13060]: 

200 data_temp[-1, 1:] += data_temp[i, 1:] 

201 to_delete.append(i) 

202 

203 data_temp = np.delete(data_temp, to_delete, 0) 

204 sorted_inds = np.argsort(data_temp[:, 0]) 

205 data_temp = data_temp[sorted_inds, :] 

206 if to_delete == []: 

207 return data 

208 else: 

209 return data_temp

281def get_age_population_data(read_data=dd.defaultDict['read_data'], 

282 file_format=dd.defaultDict['file_format'], 

283 out_folder=dd.defaultDict['out_folder'], 

284 no_raw=dd.defaultDict['no_raw'], 

285 write_df=True, 

286 merge_eisenach=True): 

287 """! Download data with age splitting 

288 

289 Data is downloaded from the following sources 

290 - our own migration [stored in "counties"] 

291 - Zensus2011 data splitted by gender for whole germany, states, counties in xls 

292 with additional data from 30.04.2011 (just used for reg_key?) [stored in "reg_key"] 

293 - Zensus2011 data from opendata splitted for age and gender [stored in "zensus"] 

294 

295 Data is either downloaded or read from "out_folder"/Germany/. 

296 

297 Working with the data includes: 

298 - For the Zensus data the male and female data is added to get just the age dependence 

299 - Population from the Zensus data of all age groups is scaled to the total population of our more recent migration 

300 data by a factor which represents the relative increase/decrease in population size 

301 between 2011 and 2019 for each county" 

302 

303 @param read_data False [Default] or True. Defines if data is read from file or downloaded. 

304 @param file_format File format which is used for writing the data. Default defined in defaultDict. 

305 @param out_folder Path to folder where data is written in folder out_folder/Germany. 

306 """ 

307 counties, reg_key, zensus = load_age_population_data(out_folder) 

308 

309 # find region keys for census population data 

310 key = np.zeros((len(zensus))) 

311 for i in range(len(key)): 

312 for j in range(len(reg_key)): 

313 if zensus.Name.values[i] == reg_key['NAME'].values.astype(str)[j]: 

314 if zensus.EWZ.values[i] == round(reg_key['Zensus_EWZ'].values[j]*1000): 

315 key[i] = reg_key['AGS'].values[j] 

316 

317 unique, inds, count = np.unique(key, return_index=True, return_counts=True) 

318 

319 male = ['M_Unter_3', 'M_3_bis_5', 'M_6_bis_14', 'M_15_bis_17', 'M_18_bis_24', 

320 'M_25_bis_29', 'M_30_bis_39', 'M_40_bis_49', 'M_50_bis_64', 

321 'M_65_bis_74', 'M_75_und_aelter'] 

322 female = ['W_Unter_3', 'W_3_bis_5', 'W_6_bis_14', 'W_15_bis_17', 'W_18_bis_24', 

323 'W_25_bis_29', 'W_30_bis_39', 'W_40_bis_49', 'W_50_bis_64', 

324 'W_65_bis_74', 'W_75_und_aelter'] 

325 columns = [dd.EngEng['idCounty'], 'Total', '<3 years', '3-5 years', '6-14 years', '15-17 years', '18-24 years', 

326 '25-29 years', '30-39 years', '40-49 years', '50-64 years', 

327 '65-74 years', '>74 years'] 

328 

329 # add male and female population data 

330 data = np.zeros((len(inds), len(male)+2)) 

331 data[:, 0] = key[inds].astype(int) 

332 data[:, 1] = zensus['EWZ'].values[inds].astype(int) 

333 for i in range(len(male)): 

334 data[:, i+2] = zensus[male[i] 

335 ].values[inds].astype(int) + zensus[female[i]].values[inds].astype(int) 

336 

337 data = get_new_counties(data) 

338 

339 # compute ratio of current and 2011 population data 

340 ratio = np.ones(len(data[:, 0])) 

341 for i in range(len(ratio)): 

342 for j in range(len(counties)): 

343 if not counties['Schlüssel-nummer'].isnull().values[j]: 

344 try: 

345 if data[i, 0] == int(counties['Schlüssel-nummer'].values[j]): 

346 ratio[i] = counties['Bevölkerung2)'].values[j]/data[i, 1] 

347 except: 

348 dummy = 0 

349 

350 # adjust population data for all ages to current level 

351 data_current = np.zeros(data.shape) 

352 data_current[:, 0] = data[:, 0].copy() 

353 for i in range(len(data[0, :]) - 1): 

354 data_current[:, i + 1] = np.multiply(data[:, i + 1], ratio) 

355 

356 # create dataframe 

357 df = pandas.DataFrame(data.astype(int), columns=columns) 

358 df_current = pandas.DataFrame( 

359 np.round(data_current).astype(int), columns=columns) 

360 df_401 = df_current.copy() 

361 df_current_401 = df_current.copy() 

362 # From official county list, merge Eisenach and 

363 # Wartburgkreis for now (as of oct. 2021) 

364 df_current = geoger.merge_df_counties_all( 

365 df_current, sorting=[dd.EngEng["idCounty"]], 

366 columns=dd.EngEng["idCounty"]) 

367 

368 df = geoger.merge_df_counties_all( 

369 df, sorting=[dd.EngEng["idCounty"]], 

370 columns=dd.EngEng["idCounty"]) 

371 

372 directory = os.path.join(out_folder, 'Germany/') 

373 gd.check_dir(directory) 

374 

375 if write_df: 

376 gd.write_dataframe(df, directory, 'county_population', file_format) 

377 gd.write_dataframe(df_current, directory, 

378 'county_current_population', file_format) 

379 # TODO there should be a more elegant version to write different version with Eisenach merged or not 

380 # or it should be prevented directly to write if Eisenach is not merged... to discuss... 

381 gd.write_dataframe(df_401, directory, 

382 'county_population_dim401', file_format) 

383 gd.write_dataframe(df_current_401, directory, 

384 'county_current_population_dim401', file_format) 

385 

386 if merge_eisenach: 

387 return df_current 

388 else: 

389 return df_current_401
