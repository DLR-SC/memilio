36def get_new_counties(data): 

37 """! Creates 7 new counties that were formed since 2011 and deletes old counties 

38 

39 Downloaded data is from 2011. 

40 However, new counties have been defined. 

41 Thus, the new ones have to be added, the data has to be calculated and the old counties have to be deleted. 

42 

43 @param data_temp Pandas dataframe 

44 @return Changed data 

45 """ 

46 

47 # create 7 new counties 

48 data_temp = np.append(data, np.zeros((7, data.shape[1])), axis=0) 

49 

50 # Göttingen 

51 data_temp[-7, 0] = 3159 

52 

53 # Mecklenburgische Seenplatte 

54 data_temp[-6, 0] = 13071 

55 

56 # Landkreis Rostock 

57 data_temp[-5, 0] = 13072 

58 

59 # Vorpommern Rügen 

60 data_temp[-4, 0] = 13073 

61 

62 # Nordwestmecklenburg 

63 data_temp[-3, 0] = 13074 

64 

65 # Vorpommern Greifswald 

66 data_temp[-2, 0] = 13075 

67 

68 # Ludwigslust-Parchim 

69 data_temp[-1, 0] = 13076 

70 

71 to_delete = [] 

72 

73 for i in range(len(data_temp[:, 0])): 

74 # fuse "Göttingen" and "Osterode am Harz" into Göttingen 

75 if data_temp[i, 0] in [3152, 3156]: 

76 data_temp[-7, 1:] += data_temp[i, 1:] 

77 to_delete.append(i) 

78 

79 # fuse "Müritz", "Neubrandenburg", "Mecklenburg-Sterlitz" 

80 # and "Demmin" into "Mecklenburgische Seenplatte" 

81 if data_temp[i, 0] in [13056, 13002, 13055, 13052]: 

82 data_temp[-6, 1:] += data_temp[i, 1:] 

83 to_delete.append(i) 

84 

85 # fuse "Bad Doberan and Güstrow" into "Landkreis Rostosck" 

86 if data_temp[i, 0] in [13051, 13053]: 

87 data_temp[-5, 1:] += data_temp[i, 1:] 

88 to_delete.append(i) 

89 

90 # fuse "Rügen", "Stralsund" and "Nordvorpommern" into 

91 # "Vorpommern Rügen" 

92 if data_temp[i, 0] in [13061, 13005, 13057]: 

93 data_temp[-4, 1:] += data_temp[i, 1:] 

94 to_delete.append(i) 

95 

96 # fuse "Wismar" and "Nordwestmecklenburg" into 

97 # "Nordwestmecklenburg" 

98 if data_temp[i, 0] in [13006, 13058]: 

99 data_temp[-3, 1:] += data_temp[i, 1:] 

100 to_delete.append(i) 

101 

102 # fuse "Ostvorpommern", "Uecker-Randow" and "Greifswald" 

103 # into "Vorpommern Greifswald" 

104 if data_temp[i, 0] in [13059, 13062, 13001]: 

105 data_temp[-2, 1:] += data_temp[i, 1:] 

106 to_delete.append(i) 

107 

108 # fuse "Ludwigslust" and "Parchim" into "Ludwigslust-Parchim" 

109 if data_temp[i, 0] in [13054, 13060]: 

110 data_temp[-1, 1:] += data_temp[i, 1:] 

111 to_delete.append(i) 

112 

113 data_temp = np.delete(data_temp, to_delete, 0) 

114 sorted_inds = np.argsort(data_temp[:, 0]) 

115 data_temp = data_temp[sorted_inds, :] 

116 return data_temp
