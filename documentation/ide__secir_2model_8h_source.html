228  // ---- Additional functionality such as constraint checking, setters and getters, etc. ----
233  bool check_constraints(ScalarType dt) const
234  {
235 
236  if (!((size_t)m_transitions.get_num_elements() == (size_t)InfectionTransition::Count * m_num_agegroups)) {
237  log_error("A variable given for model construction is not valid. Number of elements in transition vector "
238  "does not match the required number.");
239  return true;
240  }
241 
242  for (AgeGroup group = AgeGroup(0); group < AgeGroup(m_num_agegroups); ++group) {
243 
244  for (int i = 0; i < (int)InfectionState::Count; i++) {
245  int index = get_state_flat_index(i, group);
246  if (m_populations[0][index] < 0) {
247  log_error("Initialization failed. Initial values for populations are less than zero.");
248  return true;
249  }
250  }
251  }
252 
253  // It may be possible to run the simulation with fewer time points, but this number ensures that it is possible.
254  if (m_transitions.get_num_time_points() < (Eigen::Index)std::ceil(get_global_support_max(dt) / dt)) {
255  log_error("Initialization failed. Not enough time points for transitions given before start of "
256  "simulation.");
257  return true;
258  }
259 
260  for (AgeGroup group = AgeGroup(0); group < AgeGroup(m_num_agegroups); ++group) {
261 
262  for (int i = 0; i < m_transitions.get_num_time_points(); i++) {
263  for (int j = 0; j < (int)InfectionTransition::Count; j++) {
264  int index = get_transition_flat_index(j, group);
265  if (m_transitions[i][index] < 0) {
266  log_error(
267  "Initialization failed. One or more initial value for transitions is less than zero.");
268  return true;
269  }
270  }
271  }
272  }
273  if (m_transitions.get_last_time() != m_populations.get_last_time()) {
274  log_error("Last time point of TimeSeries for transitions does not match last time point of "
275  "TimeSeries for "
276  "compartments. Both of these time points have to agree for a sensible simulation.");
277  return true;
278  }
279 
280  if (m_populations.get_num_time_points() != 1) {
281  log_error("The TimeSeries for the compartments contains more than one time point. It is unclear how to "
282  "initialize.");
283  return true;
284  }
285 
286  return parameters.check_constraints();
287  }
