60  void advance(double t_max = 1.0)
61  {
62  auto dt = m_dt;
63  while (m_t < t_max) {
64  if (m_t + dt > t_max) {
65  dt = t_max - m_t;
66  }
67 
68  for (auto& n : m_graph.nodes()) {
69  m_node_func(m_t, dt, n.property);
70  }
71 
72  m_t += dt;
73

75  m_edge_func(m_t, dt, e.property, m_graph.nodes()[e.start_node_idx].property, m_graph.nodes()[e.end_node_idx].property);
76  }
77  }
