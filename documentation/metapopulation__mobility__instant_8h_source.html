417 template <typename FP>
418 template <class Sim>
419 void MobilityEdge<FP>::apply_mobility(FP t, FP dt, SimulationNode<Sim>& node_from, SimulationNode<Sim>& node_to)
420 {
421  //check dynamic npis
422  if (m_t_last_dynamic_npi_check == -std::numeric_limits<double>::infinity()) {
423  m_t_last_dynamic_npi_check = node_from.get_t0();
424  }
425 
426  auto& dyn_npis = m_parameters.get_dynamic_npis_infected();
427  if (dyn_npis.get_thresholds().size() > 0 &&
428  floating_point_greater_equal(t, m_t_last_dynamic_npi_check + dyn_npis.get_interval().get())) {
429  auto inf_rel = get_infections_relative(node_from, t, node_from.get_last_state()) * dyn_npis.get_base_value();
430  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
431  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
432  (exceeded_threshold->first > m_dynamic_npi.first ||
433  t > double(m_dynamic_npi.second))) { //old NPI was weaker or is expired
434  auto t_end = SimulationTime(t + double(dyn_npis.get_duration()));
435  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
436  implement_dynamic_npis(
437  m_parameters.get_coefficients(), exceeded_threshold->second, SimulationTime(t), t_end, [this](auto& g) {
438  return make_mobility_damping_vector(m_parameters.get_coefficients().get_shape(), g);
439  });
440  }
441  m_t_last_dynamic_npi_check = t;
442  }
443 
444  //returns
445  for (Eigen::Index i = m_return_times.get_num_time_points() - 1; i >= 0; --i) {
446  if (m_return_times.get_time(i) <= t) {
447  auto v0 = find_value_reverse(node_to.get_result(), m_mobile_population.get_time(i), 1e-10, 1e-10);
448  assert(v0 != node_to.get_result().rend() && "unexpected error.");
449  calculate_mobility_returns<FP, Sim>(m_mobile_population[i], node_to.get_simulation(), *v0,
450  m_mobile_population.get_time(i), dt);
451 
452  //the lower-order return calculation may in rare cases produce negative compartments,
453  //especially at the beginning of the simulation.
454  //fix by subtracting the supernumerous returns from the biggest compartment of the age group.
455  Eigen::VectorXd remaining_after_return =
456  (node_to.get_result().get_last_value() - m_mobile_population[i]).eval();
457  for (Eigen::Index j = 0; j < node_to.get_result().get_last_value().size(); ++j) {
458  if (remaining_after_return(j) < 0) {
459  auto num_comparts = (Eigen::Index)Sim::Model::Compartments::Count;
460  auto group = Eigen::Index(j / num_comparts);
461  auto compart = j % num_comparts;
462  log(remaining_after_return(j) < -1e-3 ? LogLevel::warn : LogLevel::info,
463  "Underflow during mobility returns at time {}, compartment {}, age group {}: {}", t, compart,
464  group, remaining_after_return(j));
465  Eigen::Index max_index;
466  slice(remaining_after_return, {group * num_comparts, num_comparts}).maxCoeff(&max_index);
467  log_info("Transferring to compartment {}", max_index);
468  max_index += group * num_comparts;
469  m_mobile_population[i](max_index) -= remaining_after_return(j);
470  m_mobile_population[i](j) += remaining_after_return(j);
471  }
472  }
473  node_from.get_result().get_last_value() += m_mobile_population[i];
474  node_to.get_result().get_last_value() -= m_mobile_population[i];
475  m_mobile_population.remove_time_point(i);
476  m_return_times.remove_time_point(i);
477  }
478  }
479 
480  if (!m_return_mobile_population && (m_parameters.get_coefficients().get_matrix_at(t).array() > 0.0).any()) {
481  //normal daily mobility
482  m_mobile_population.add_time_point(
483  t, (node_from.get_last_state().array() * m_parameters.get_coefficients().get_matrix_at(t).array() *
484  get_mobility_factors(node_from, t, node_from.get_last_state()).array())
485  .matrix());
486  m_return_times.add_time_point(t + dt);
487 
488  test_commuters(node_from, m_mobile_population.get_last_value(), t);
489 
490  node_to.get_result().get_last_value() += m_mobile_population.get_last_value();
491  node_from.get_result().get_last_value() -= m_mobile_population.get_last_value();
492  }
493  m_return_mobile_population = !m_return_mobile_population;
494 }
