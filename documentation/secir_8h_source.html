37 class SecirModel : public CompartmentalModel<InfectionState, Populations<AgeGroup, InfectionState>, SecirParams>
38 {
39  using Base = CompartmentalModel<InfectionState, mio::Populations<AgeGroup, InfectionState>, SecirParams>;
40  using Pa = Base::ParameterSet;
41  using Po = Base::Populations;
42 
43 public:
44  SecirModel(const Populations& pop, const ParameterSet& params)
45  : Base(pop, params)
46  {
47 #if !USE_DERIV_FUNC
48  size_t n_agegroups = (size_t)AgeGroup::Count;
49  for (size_t i = 0; i < n_agegroups; i++) {
50  for (size_t j = 0; j < n_agegroups; j++) {
51 
52  // Si to Ei individually for each age group j
53  this->add_flow(
54  std::make_tuple((AgeGroup)i, InfectionState::S), std::make_tuple((AgeGroup)i, InfectionState::E),
55  [i, j](Pa const& p, Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y,
56  double t) {
57  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
58  auto test_and_trace_required = (1 - params.probabilities[i].get_asymp_per_infectious()) *
59  dummy_R3 * Po::get_from(pop, (AgeGroup)i, InfectionState::C);
60  auto risk_from_symptomatic =
61  smoother_cosine(test_and_trace_required, params.get_test_and_trace_capacity(),
62  params.get_test_and_trace_capacity() * 5,
63  params.probabilities[i].get_risk_from_symptomatic(),
64  params.probabilities[i].get_test_and_trace_max_risk_from_symptomatic());
65 
66  // effective contact rate by contact rate between groups i and j and damping j
67  ScalarType season_val =
68  (1 + p.get_seasonality() * sin(3.141592653589793 *
69  (std::fmod((p.get_start_day() + t), 365.0) / 182.5 + 0.5)));
70  ScalarType cont_freq_eff =
71  season_val *
72  contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>(i), static_cast<Eigen::Index>(j));
73  ScalarType Nj = Po::get_from(pop, (AgeGroup)j, InfectionState::S) +
74  Po::get_from(pop, (AgeGroup)j, InfectionState::E) +
75  Po::get_from(pop, (AgeGroup)j, InfectionState::C) +
76  Po::get_from(pop, (AgeGroup)j, InfectionState::I) +
77  Po::get_from(pop, (AgeGroup)j, InfectionState::H) +
78  Po::get_from(pop, (AgeGroup)j, InfectionState::U) +
79  Po::get_from(pop, (AgeGroup)j, InfectionState::R); // without died people
80  ScalarType divNj = 1.0 / Nj; // precompute 1.0/Nj
81  ScalarType Si = Po::get_from(y, (AgeGroup)i, InfectionState::S);
82  ScalarType Cj = Po::get_from(pop, (AgeGroup)j, InfectionState::C);
83  ScalarType Ij = Po::get_from(pop, (AgeGroup)j, InfectionState::I);
84  return Si * cont_freq_eff * divNj * p.probabilities[i].get_infection_from_contact() *
85  (p.probabilities[j].get_carrier_infectability() * Cj + risk_from_symptomatic * Ij);
86  });
87  }
88 
89  // Ei to Ci
90  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::E),
91  std::make_tuple((AgeGroup)i, InfectionState::C),
92  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/,
93  Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
94  return Po::get_from(y, (AgeGroup)i, InfectionState::E) /
95  (2 * p.times[i].get_serialinterval() - p.times[i].get_incubation());
96  });
97 
98  // Ci to Ii
99  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::C),
100  std::make_tuple((AgeGroup)i, InfectionState::I),
101  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/,
102  Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
103  double dummy_R3 = 0.5 / (p.times[i].get_incubation() - p.times[i].get_serialinterval());
104  double alpha = p.probabilities[i].get_asymp_per_infectious();
105  return ((1 - alpha) * dummy_R3) * Po::get_from(y, (AgeGroup)i, InfectionState::C);
106  });
107 
108  // Ci to Ri
109  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::C),
110  std::make_tuple((AgeGroup)i, InfectionState::R),
111  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/,
112  Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
113  double alpha = p.probabilities[i].get_asymp_per_infectious();
114  return (alpha / p.times[i].get_infectious_asymp()) *
115  Po::get_from(y, (AgeGroup)i, InfectionState::C);
116  });
117 
118  // Ii to Ri
119  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::I),
120  std::make_tuple((AgeGroup)i, InfectionState::R),
121  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/,
122  Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
123  return Po::get_from(y, (AgeGroup)i, InfectionState::I) *
124  (1 - p.probabilities[i].get_hospitalized_per_infectious()) /
125  p.times[i].get_infectious_mild();
126  });
127 
128  // Ii to Hi
129  this->add_flow(
130  std::make_tuple((AgeGroup)i, InfectionState::I), std::make_tuple((AgeGroup)i, InfectionState::H),
131  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y,
132  double /*t*/) {
133  return Po::get_from(y, (AgeGroup)i, InfectionState::I) *
134  p.probabilities[i].get_hospitalized_per_infectious() / p.times[i].get_home_to_hospitalized();
135  });
136 
137  // Hi to Ui
138  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::H),
139  std::make_tuple((AgeGroup)i, InfectionState::U),
140  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/,
141  Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
142  ScalarType icu_occupancy = 0;
143  for (size_t j = 0; j < (size_t)AgeGroup::Count; ++j) {
144  icu_occupancy += Po::get_from(y, (AgeGroup)j, InfectionState::U);
145  }
146 
147  ScalarType prob_hosp2icu =
148  smoother_cosine(icu_occupancy, 0.90 * p.get_icu_capacity(), p.get_icu_capacity(),
149  p.probabilities[i].get_icu_per_hospitalized(), 0);
150 
151  return Po::get_from(y, (AgeGroup)i, InfectionState::H) * prob_hosp2icu /
152  p.times[i].get_hospitalized_to_icu();
153  });
154 
155  // Hi to Di
156  this->add_flow(
157  std::make_tuple((AgeGroup)i, InfectionState::H), std::make_tuple((AgeGroup)i, InfectionState::D),
158  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y,
159  double /*t*/) {
160  ScalarType icu_occupancy = 0;
161  for (size_t j = 0; j < (size_t)AgeGroup::Count; ++j) {
162  icu_occupancy += Po::get_from(y, (AgeGroup)j, InfectionState::U);
163  }
164 
165  ScalarType prob_hosp2icu =
166  smoother_cosine(icu_occupancy, 0.90 * p.get_icu_capacity(), p.get_icu_capacity(),
167  p.probabilities[i].get_icu_per_hospitalized(), 0);
168  ScalarType prob_hosp2dead = p.probabilities[i].get_icu_per_hospitalized() - prob_hosp2icu;
169 
170  return Po::get_from(y, (AgeGroup)i, InfectionState::H) * prob_hosp2dead /
171  p.times[i].get_hospitalized_to_icu();
172  });
173 
174  // Hi to Ri
175  this->add_flow(
176  std::make_tuple((AgeGroup)i, InfectionState::H), std::make_tuple((AgeGroup)i, InfectionState::R),
177  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/, Eigen::Ref<const Eigen::VectorXd> y,
178  double /*t*/) {
179  return Po::get_from(y, (AgeGroup)i, InfectionState::H) *
180  (1 - p.probabilities[i].get_icu_per_hospitalized()) / p.times[i].get_hospitalized_to_home();
181  });
182 
183  // Ui to Ri
184  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::U),
185  std::make_tuple((AgeGroup)i, InfectionState::R),
186  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/,
187  Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
188  return Po::get_from(y, (AgeGroup)i, InfectionState::U) *
189  (1 - p.probabilities[i].get_dead_per_icu()) / p.times[i].get_icu_to_home();
190  });
191 
192  // Ui to Di
193  this->add_flow(std::make_tuple((AgeGroup)i, InfectionState::U),
194  std::make_tuple((AgeGroup)i, InfectionState::D),
195  [i](Pa const& p, Eigen::Ref<const Eigen::VectorXd> /*pop*/,
196  Eigen::Ref<const Eigen::VectorXd> y, double /*t*/) {
197  return Po::get_from(y, (AgeGroup)i, InfectionState::U) *
198  p.probabilities[i].get_dead_per_icu() / p.times[i].get_icu_to_dead();
199  });
200  }
201 #endif

211  void get_derivatives(Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, double t,
212  Eigen::Ref<Eigen::VectorXd> dydt) const override
213  {
214  // alpha // percentage of asymptomatic cases
215  // beta // risk of infection from the infected symptomatic patients
216  // rho // hospitalized per infectious
217  // theta // icu per hospitalized
218  // delta // deaths per ICUs
219  // 0: S, 1: E, 2: C, 3: I, 4: H, 5: U, 6: R, 7: D
220  auto const& params = this->parameters;
221  AgeGroup n_agegroups = params.get_num_groups();
222 
223  ContactMatrixGroup const& contact_matrix = params.get<mio::ContactPatterns>();
224 
225  auto icu_occupancy = 0.0;
226  auto test_and_trace_required = 0.0;
227  for (auto i = AgeGroup(0); i < n_agegroups; ++i) {
228  auto dummy_R3 = 0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]);
229  test_and_trace_required += (1 - params.get<AsymptomaticCasesPerInfectious>()[i]) * dummy_R3 *
230  this->populations.get_from(pop, {i, InfectionState::Carrier});
231  icu_occupancy += this->populations.get_from(pop, {i, InfectionState::ICU});
232  }
233 
234  for (auto i = AgeGroup(0); i < n_agegroups; i++) {
235 
236  size_t Si = this->populations.get_flat_index({i, InfectionState::Susceptible});
237  size_t Ei = this->populations.get_flat_index({i, InfectionState::Exposed});
238  size_t Ci = this->populations.get_flat_index({i, InfectionState::Carrier});
239  size_t Ii = this->populations.get_flat_index({i, InfectionState::Infected});
240  size_t Hi = this->populations.get_flat_index({i, InfectionState::Hospitalized});
241  size_t Ui = this->populations.get_flat_index({i, InfectionState::ICU});
242  size_t Ri = this->populations.get_flat_index({i, InfectionState::Recovered});
243  size_t Di = this->populations.get_flat_index({i, InfectionState::Dead});
244 
245  dydt[Si] = 0;
246  dydt[Ei] = 0;
247 
248  double dummy_R2 =
249  1.0 / (2 * params.get<SerialInterval>()[i] - params.get<IncubationTime>()[i]); // R2 = 1/(2SI-TINC)
250  double dummy_R3 =
251  0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]); // R3 = 1/(2(TINC-SI))
252 
253  for (auto j = AgeGroup(0); j < n_agegroups; j++) {
254  size_t Sj = this->populations.get_flat_index({j, InfectionState::Susceptible});
255  size_t Ej = this->populations.get_flat_index({j, InfectionState::Exposed});
256  size_t Cj = this->populations.get_flat_index({j, InfectionState::Carrier});
257  size_t Ij = this->populations.get_flat_index({j, InfectionState::Infected});
258  size_t Hj = this->populations.get_flat_index({j, InfectionState::Hospitalized});
259  size_t Uj = this->populations.get_flat_index({j, InfectionState::ICU});
260  size_t Rj = this->populations.get_flat_index({j, InfectionState::Recovered});
261 
262  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
263  auto risk_from_symptomatic = smoother_cosine(
264  test_and_trace_required, params.get<TestAndTraceCapacity>(), params.get<TestAndTraceCapacity>() * 5,
265  params.get<RiskOfInfectionFromSymptomatic>()[j], params.get<MaxRiskOfInfectionFromSymptomatic>()[j]);
266 
267  // effective contact rate by contact rate between groups i and j and damping j
268  double season_val = (1 + params.get<mio::Seasonality>() *
269  sin(3.141592653589793 *
270  (std::fmod((params.get<mio::StartDay>() + t), 365.0) / 182.5 + 0.5)));
271  double cont_freq_eff =
272  season_val * contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>((size_t)i),
273  static_cast<Eigen::Index>((size_t)j));
274  double Nj = pop[Sj] + pop[Ej] + pop[Cj] + pop[Ij] + pop[Hj] + pop[Uj] + pop[Rj]; // without died people
275  double divNj = 1.0 / Nj; // precompute 1.0/Nj
276  double dummy_S =
277  y[Si] * cont_freq_eff * divNj * params.get<InfectionProbabilityFromContact>()[i] *
278  (params.get<RelativeCarrierInfectability>()[j] * pop[Cj] + risk_from_symptomatic * pop[Ij]);
279 
280  dydt[Si] -= dummy_S; // -R1*(C+beta*I)*S/N0
281  dydt[Ei] += dummy_S; // R1*(C+beta*I)*S/N0-R2*E
282  }
283 
284  // ICU capacity shortage is close
285  double prob_hosp2icu =
286  smoother_cosine(icu_occupancy, 0.90 * params.get<mio::ICUCapacity>(), params.get<mio::ICUCapacity>(),
287  params.get<ICUCasesPerHospitalized>()[i], 0);
288 
289  double prob_hosp2dead = params.get<ICUCasesPerHospitalized>()[i] - prob_hosp2icu;
290 
291  dydt[Ei] -= dummy_R2 * y[Ei]; // only exchange of E and C done here
292  dydt[Ci] = dummy_R2 * y[Ei] -
293  ((1 - params.get<AsymptomaticCasesPerInfectious>()[i]) * dummy_R3 +
294  params.get<AsymptomaticCasesPerInfectious>()[i] / params.get<InfectiousTimeAsymptomatic>()[i]) *
295  y[Ci];
296  dydt[Ii] = (1 - params.get<AsymptomaticCasesPerInfectious>()[i]) * dummy_R3 * y[Ci] -
297  ((1 - params.get<HospitalizedCasesPerInfectious>()[i]) / params.get<InfectiousTimeMild>()[i] +
298  params.get<HospitalizedCasesPerInfectious>()[i] / params.get<HomeToHospitalizedTime>()[i]) *
299  y[Ii];
300  dydt[Hi] =
301  params.get<HospitalizedCasesPerInfectious>()[i] / params.get<HomeToHospitalizedTime>()[i] * y[Ii] -
302  ((1 - params.get<ICUCasesPerHospitalized>()[i]) / params.get<HospitalizedToHomeTime>()[i] +
303  params.get<ICUCasesPerHospitalized>()[i] / params.get<HospitalizedToICUTime>()[i]) *
304  y[Hi];
305  dydt[Ui] = -((1 - params.get<DeathsPerICU>()[i]) / params.get<ICUToHomeTime>()[i] +
306  params.get<DeathsPerICU>()[i] / params.get<ICUToDeathTime>()[i]) *
307  y[Ui];
308  // add flow from hosp to icu according to potentially adjusted probability due to ICU limits
309  dydt[Ui] += prob_hosp2icu / params.get<HospitalizedToICUTime>()[i] * y[Hi];
310 
311  dydt[Ri] =
312  params.get<AsymptomaticCasesPerInfectious>()[i] / params.get<InfectiousTimeAsymptomatic>()[i] * y[Ci] +
313  (1 - params.get<HospitalizedCasesPerInfectious>()[i]) / params.get<InfectiousTimeMild>()[i] * y[Ii] +
314  (1 - params.get<ICUCasesPerHospitalized>()[i]) / params.get<HospitalizedToHomeTime>()[i] * y[Hi] +
315  (1 - params.get<DeathsPerICU>()[i]) / params.get<ICUToHomeTime>()[i] * y[Ui];
316 
317  dydt[Di] = params.get<DeathsPerICU>()[i] / params.get<ICUToDeathTime>()[i] * y[Ui];
318  // add potential, additional deaths due to ICU overflow
319  dydt[Di] += prob_hosp2dead / params.get<HospitalizedToICUTime>()[i] * y[Hi];
320  }
321  }

375 template <class Base>
376 class SecirSimulation : public Base
377 {
378 public:
385  SecirSimulation(SecirModel const& model, double t0 = 0., double dt = 0.1)
386  : Base(model, t0, dt)
387  , m_t_last_npi_check(t0)
388  {
389  }
390 
398  Eigen::Ref<Eigen::VectorXd> advance(double tmax)
399  {
400  auto& dyn_npis = this->get_model().parameters.template get<DynamicNPIsInfected>();
401  auto& contact_patterns = this->get_model().parameters.template get<ContactPatterns>();
402  if (dyn_npis.get_thresholds().size() > 0) {
403  auto t = Base::get_result().get_last_time();
404  const auto dt = dyn_npis.get_interval().get();
405 
406  while (t < tmax) {
407  auto dt_eff = std::min({dt, tmax - t, m_t_last_npi_check + dt - t});
408 
409  Base::advance(t + dt_eff);
410  t = t + dt_eff;
411 
412  if (floating_point_greater_equal(t, m_t_last_npi_check + dt)) {
413  auto inf_rel = get_infections_relative(*this, t, this->get_result().get_last_value()) *
414  dyn_npis.get_base_value();
415  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
416  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
417  (exceeded_threshold->first > m_dynamic_npi.first ||
418  t > double(m_dynamic_npi.second))) { //old npi was weaker or is expired
419  auto t_end = mio::SimulationTime(t + double(dyn_npis.get_duration()));
420  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
421  mio::implement_dynamic_npis(contact_patterns.get_cont_freq_mat(), exceeded_threshold->second,
422  SimulationTime(t), t_end, [](auto& g) {
423  return mio::make_contact_damping_matrix(g);
424  });
425  }
426 
427  m_t_last_npi_check = t;
428  }
429  }
430 
431  return this->get_result().get_last_value();
432  }
433  else {
434  return Base::advance(tmax);
435  }
436  }
437 
438 private:
439  double m_t_last_npi_check;
440  std::pair<double, SimulationTime> m_dynamic_npi = {-std::numeric_limits<double>::max(), mio::SimulationTime(0)};
441 };

481 template <class Base = Simulation<SecirModel>>
482 auto get_migration_factors(const SecirSimulation<Base>& sim, double /*t*/, const Eigen::Ref<const Eigen::VectorXd>& y)
483 {
484  auto& params = sim.get_model().parameters;
485  //parameters as arrays
486  auto&& t_inc = params.template get<IncubationTime>().array().template cast<double>();
487  auto&& t_ser = params.template get<SerialInterval>().array().template cast<double>();
488  auto&& p_asymp = params.template get<AsymptomaticCasesPerInfectious>().array().template cast<double>();
489  auto&& p_inf = params.template get<RiskOfInfectionFromSymptomatic>().array().template cast<double>();
490  auto&& p_inf_max = params.template get<MaxRiskOfInfectionFromSymptomatic>().array().template cast<double>();
491  //slice of carriers
492  auto y_car = slice(y, {Eigen::Index(InfectionState::Carrier), Eigen::Index(size_t(params.get_num_groups())),
493  Eigen::Index(InfectionState::Count)});
494 
495  //compute isolation, same as infection risk from main model
496  auto R3 = 0.5 / (t_inc - t_ser);
497  auto test_and_trace_required = ((1 - p_asymp) * R3 * y_car.array()).sum();
498  auto test_and_trace_capacity = double(params.template get<TestAndTraceCapacity>());
