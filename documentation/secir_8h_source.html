56  void get_derivatives(Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, double t,
57  Eigen::Ref<Eigen::VectorXd> dydt) const override
58  {
59  // alpha // percentage of asymptomatic cases
60  // beta // risk of infection from the infected symptomatic patients
61  // rho // hospitalized per infectious
62  // theta // icu per hospitalized
63  // delta // deaths per ICUs
64  // 0: S, 1: E, 2: C, 3: I, 4: H, 5: U, 6: R, 7: D
65  auto const& params = this->parameters;
66  AgeGroup n_agegroups = params.get_num_groups();
67 
68  ContactMatrixGroup const& contact_matrix = params.get<mio::ContactPatterns>();
69 
70  auto icu_occupancy = 0.0;
71  auto test_and_trace_required = 0.0;
72  for (auto i = AgeGroup(0); i < n_agegroups; ++i) {
73  auto rateINS = 0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]);
74  test_and_trace_required += (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * rateINS *
75  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptoms});
76  icu_occupancy += this->populations.get_from(pop, {i, InfectionState::InfectedCritical});
77  }
78 
79  for (auto i = AgeGroup(0); i < n_agegroups; i++) {
80 
81  size_t Si = this->populations.get_flat_index({i, InfectionState::Susceptible});
82  size_t Ei = this->populations.get_flat_index({i, InfectionState::Exposed});
83  size_t INSi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptoms});
84  size_t ISyi = this->populations.get_flat_index({i, InfectionState::InfectedSymptoms});
85  size_t ISevi = this->populations.get_flat_index({i, InfectionState::InfectedSevere});
86  size_t ICri = this->populations.get_flat_index({i, InfectionState::InfectedCritical});
87  size_t Ri = this->populations.get_flat_index({i, InfectionState::Recovered});
88  size_t Di = this->populations.get_flat_index({i, InfectionState::Dead});
89 
90  dydt[Si] = 0;
91  dydt[Ei] = 0;
92 
93  double rateE =
94  1.0 / (2 * params.get<SerialInterval>()[i] - params.get<IncubationTime>()[i]); // R2 = 1/(2SI-TINC)
95  double rateINS =
96  0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]); // R3 = 1/(2(TINC-SI))
97 
98  for (auto j = AgeGroup(0); j < n_agegroups; j++) {
99  size_t Sj = this->populations.get_flat_index({j, InfectionState::Susceptible});
100  size_t Ej = this->populations.get_flat_index({j, InfectionState::Exposed});
101  size_t INSj = this->populations.get_flat_index({j, InfectionState::InfectedNoSymptoms});
102  size_t ISyj = this->populations.get_flat_index({j, InfectionState::InfectedSymptoms});
103  size_t ISevj = this->populations.get_flat_index({j, InfectionState::InfectedSevere});
104  size_t ICrj = this->populations.get_flat_index({j, InfectionState::InfectedCritical});
105  size_t Rj = this->populations.get_flat_index({j, InfectionState::Recovered});
106 
107  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
108  auto riskFromInfectedSymptomatic = smoother_cosine(
109  test_and_trace_required, params.get<TestAndTraceCapacity>(), params.get<TestAndTraceCapacity>() * 5,
110  params.get<RiskOfInfectionFromSymptomatic>()[j],
111  params.get<MaxRiskOfInfectionFromSymptomatic>()[j]);
112 
113  // effective contact rate by contact rate between groups i and j and damping j
114  double season_val = (1 + params.get<mio::Seasonality>() *
115  sin(3.141592653589793 *
116  (std::fmod((params.get<mio::StartDay>() + t), 365.0) / 182.5 + 0.5)));
117  double cont_freq_eff =
118  season_val * contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>((size_t)i),
119  static_cast<Eigen::Index>((size_t)j));
120  double Nj =
121  pop[Sj] + pop[Ej] + pop[INSj] + pop[ISyj] + pop[ISevj] + pop[ICrj] + pop[Rj]; // without died people
122  double divNj = 1.0 / Nj; // precompute 1.0/Nj
123  double dummy_S = y[Si] * cont_freq_eff * divNj * params.get<TransmissionProbabilityOnContact>()[i] *
124  (params.get<RelativeTransmissionNoSymptoms>()[j] * pop[INSj] +
125  riskFromInfectedSymptomatic * pop[ISyj]);
126 
127  dydt[Si] -= dummy_S;
128  dydt[Ei] += dummy_S;
129  }
130 
131  // ICU capacity shortage is close
132  double criticalPerSevereAdjusted =
133  smoother_cosine(icu_occupancy, 0.90 * params.get<mio::ICUCapacity>(), params.get<mio::ICUCapacity>(),
134  params.get<CriticalPerSevere>()[i], 0);
135 
136  double deathsPerSevereAdjusted = params.get<CriticalPerSevere>()[i] - criticalPerSevereAdjusted;
137 
138  dydt[Ei] -= rateE * y[Ei]; // only exchange of E and INS done here
139  dydt[INSi] = rateE * y[Ei] - rateINS * y[INSi];
140  dydt[ISyi] = (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * rateINS * y[INSi] -
141  (1 / params.get<TimeInfectedSymptoms>()[i]) * y[ISyi];
142  dydt[ISevi] = params.get<SeverePerInfectedSymptoms>()[i] / params.get<TimeInfectedSymptoms>()[i] * y[ISyi] -
143  (1 / params.get<TimeInfectedSevere>()[i]) * y[ISevi];
144  dydt[ICri] = -(1 / params.get<TimeInfectedCritical>()[i]) * y[ICri];
145  // add flow from hosp to icu according to potentially adjusted probability due to ICU limits
146  dydt[ICri] += criticalPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevi];
147 
148  dydt[Ri] =
149  params.get<RecoveredPerInfectedNoSymptoms>()[i] * rateINS * y[INSi] +
150  (1 - params.get<SeverePerInfectedSymptoms>()[i]) / params.get<TimeInfectedSymptoms>()[i] * y[ISyi] +
151  (1 - params.get<CriticalPerSevere>()[i]) / params.get<TimeInfectedSevere>()[i] * y[ISevi] +
152  (1 - params.get<DeathsPerCritical>()[i]) / params.get<TimeInfectedCritical>()[i] * y[ICri];
153 
154  dydt[Di] = params.get<DeathsPerCritical>()[i] / params.get<TimeInfectedCritical>()[i] * y[ICri];
155  // add potential, additional deaths due to ICU overflow
156  dydt[Di] += deathsPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevi];
157  }
158  }

235  Eigen::Ref<Eigen::VectorXd> advance(double tmax)
236  {
237  auto& dyn_npis = this->get_model().parameters.template get<DynamicNPIsInfectedSymptoms>();
238  auto& contact_patterns = this->get_model().parameters.template get<ContactPatterns>();
239  if (dyn_npis.get_thresholds().size() > 0) {
240  auto t = Base::get_result().get_last_time();
241  const auto dt = dyn_npis.get_interval().get();
242 
243  while (t < tmax) {
244  auto dt_eff = std::min({dt, tmax - t, m_t_last_npi_check + dt - t});
245 
246  Base::advance(t + dt_eff);
247  t = t + dt_eff;
248 
249  if (floating_point_greater_equal(t, m_t_last_npi_check + dt)) {
250  auto inf_rel = get_infections_relative(*this, t, this->get_result().get_last_value()) *
251  dyn_npis.get_base_value();
252  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
253  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
254  (exceeded_threshold->first > m_dynamic_npi.first ||
255  t > double(m_dynamic_npi.second))) { //old npi was weaker or is expired
256  auto t_end = mio::SimulationTime(t + double(dyn_npis.get_duration()));
257  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
258  mio::implement_dynamic_npis(contact_patterns.get_cont_freq_mat(), exceeded_threshold->second,
259  SimulationTime(t), t_end, [](auto& g) {
260  return mio::make_contact_damping_matrix(g);
261  });
262  }
263 
264  m_t_last_npi_check = t;
265  }
266  }
267 
268  return this->get_result().get_last_value();
269  }
270  else {
271  return Base::advance(tmax);
272  }
273  }
274 
275 private:
276  double m_t_last_npi_check;
277  std::pair<double, SimulationTime> m_dynamic_npi = {-std::numeric_limits<double>::max(), mio::SimulationTime(0)};
278 };

318 template <class Base = Simulation<SecirModel>>
319 auto get_migration_factors(const SecirSimulation<Base>& sim, double /*t*/, const Eigen::Ref<const Eigen::VectorXd>& y)
320 {
321  auto& params = sim.get_model().parameters;
322  //parameters as arrays
323  auto&& t_inc = params.template get<IncubationTime>().array().template cast<double>();
324  auto&& t_ser = params.template get<SerialInterval>().array().template cast<double>();
325  auto&& p_asymp = params.template get<RecoveredPerInfectedNoSymptoms>().array().template cast<double>();
326  auto&& p_inf = params.template get<RiskOfInfectionFromSymptomatic>().array().template cast<double>();
327  auto&& p_inf_max = params.template get<MaxRiskOfInfectionFromSymptomatic>().array().template cast<double>();
328  //slice of InfectedNoSymptoms
329  auto y_car = slice(y, {Eigen::Index(InfectionState::InfectedNoSymptoms),
330  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)});
331 
332  //compute isolation, same as infection risk from main model
333  auto R3 = 0.5 / (t_inc - t_ser);
334  auto test_and_trace_required = ((1 - p_asymp) * R3 * y_car.array()).sum();
335  auto test_and_trace_capacity = double(params.template get<TestAndTraceCapacity>());
336  auto riskFromInfectedSymptomatic = smoother_cosine(test_and_trace_required, test_and_trace_capacity,
337  test_and_trace_capacity * 5, p_inf.matrix(), p_inf_max.matrix());
338 
339  //set factor for infected
340  auto factors = Eigen::VectorXd::Ones(y.rows()).eval();
341  slice(factors, {Eigen::Index(InfectionState::InfectedSymptoms), Eigen::Index(size_t(params.get_num_groups())),
342  Eigen::Index(InfectionState::Count)})
343  .array() = riskFromInfectedSymptomatic;
344  return factors;
345 }
