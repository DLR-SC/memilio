255 template <class DampingExpression, class DampingSamplings, class F>
256 void apply_dampings(DampingExpression& damping_expression, const DampingSamplings& dampings, F make_matrix)
257 {
258  for (auto& d : dampings) {
259  for (auto& i : d.get_matrix_indices()) {
260  auto m = make_matrix(double(d.get_value()) * d.get_group_weights());
261  damping_expression[i].add_damping(m, d.get_level(), d.get_type(), d.get_time());
262  }
263  }

279 template <class V>
280 auto make_contact_damping_matrix(V&& groups)
281 {
282  auto ones_v = std::decay_t<V>::PlainObject::Constant(groups.size(), 1.0);
283  auto prod = (ones_v - groups) * (ones_v.transpose() - groups.transpose());
284  auto ones_m = decltype(prod)::PlainObject::Constant(groups.size(), groups.size(), 1.0);
285  return ones_m - sqrt(prod.array()).matrix();
286 }
