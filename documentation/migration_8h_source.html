383 template <class Sim>
384 void MigrationEdge::apply_migration(double t, double dt, SimulationNode<Sim>& node_from, SimulationNode<Sim>& node_to)
385 {
386  //check dynamic npis
387  if (m_t_last_dynamic_npi_check == -std::numeric_limits<double>::infinity()) {
388  m_t_last_dynamic_npi_check = node_from.get_t0();
389  }
390 
391  auto& dyn_npis = m_parameters.get_dynamic_npis_infected();
392  if (dyn_npis.get_thresholds().size() > 0 && floating_point_greater_equal(t, m_t_last_dynamic_npi_check + dyn_npis.get_interval().get())) {
393  auto inf_rel = get_infections_relative(node_from, t, node_from.get_last_state()) * dyn_npis.get_base_value();
394  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
395  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
396  (exceeded_threshold->first > m_dynamic_npi.first ||
397  t > double(m_dynamic_npi.second))) { //old NPI was weaker or is expired
398  auto t_end = epi::SimulationTime(t + double(dyn_npis.get_duration()));
399  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
400  epi::implement_dynamic_npis(
401  m_parameters.get_coefficients(), exceeded_threshold->second, SimulationTime(t), t_end, [this](auto& g) {
402  return epi::make_migration_damping_vector(m_parameters.get_coefficients().get_shape(), g);
403  });
404  }
405  m_t_last_dynamic_npi_check = t;
406  }
407 
408  //returns
409  for (Eigen::Index i = m_return_times.get_num_time_points() - 1; i >= 0; --i) {
410  if (m_return_times.get_time(i) <= t) {
411  auto v0 = find_value_reverse(node_to.get_result(), m_migrated.get_time(i), 1e-10, 1e-10);
412  assert(v0 != node_to.get_result().rend() && "unexpected error.");
413  calculate_migration_returns(m_migrated[i], node_to.get_simulation(), *v0, m_migrated.get_time(i), dt);
414  node_from.get_result().get_last_value() += m_migrated[i];
415  node_to.get_result().get_last_value() -= m_migrated[i];
416  m_migrated.remove_time_point(i);
417  m_return_times.remove_time_point(i);
418  }
419  }
420 
421  if (!m_return_migrated && (m_parameters.get_coefficients().get_matrix_at(t).array() > 0.0).any()) {
422  //normal daily migration
423  m_migrated.add_time_point(
424  t, (node_from.get_last_state().array() * m_parameters.get_coefficients().get_matrix_at(t).array() *
425  get_migration_factors(node_from, t, node_from.get_last_state()).array())
426  .matrix());
427  m_return_times.add_time_point(t + dt);
428 
429  node_to.get_result().get_last_value() += m_migrated.get_last_value();
430  node_from.get_result().get_last_value() -= m_migrated.get_last_value();
431  }
432  m_return_migrated = !m_return_migrated;
433 }
