574 template <class IOContext, class F, class... T>
575 details::ApplyResultT<F, T...> apply(IOContext& io, F f, const IOResult<T>&... rs)
576 {
577  //store the status of each argument in an array to check them.
578  //it would be possible to write a recursive template that checks one argument
579  //after the other without copying each status. This would probably be slightly faster
580  //and easier to optimize for the compiler. but gdb crashes trying to resolve the
581  //very long symbol in case of many arguments. Successful results are very cheap to copy
582  //and slightly worse performance in the case of an error is probably acceptable.
583 
584  //check for errors in the arguments
585  IOStatus status[] = {(rs ? IOStatus{} : rs.error())...};
586  auto iter_err = std::find_if(std::begin(status), std::end(status), [](auto& s) {
587  return s.is_error();
588  });
589 
590  //evaluate f if all succesful
591  auto result =
592  iter_err == std::end(status) ? details::eval(f, rs...) : details::ApplyResultT<F, T...>(failure(*iter_err));
593 
594  if (!result) {
595  //save error in context
596  //if the error was generated by the applied function, the context hasn't seen it yet.
597  //this allows the context to fail fast and not continue to parse more.
598  io.set_error(result.error());
599  }
600  return result;
601 }

619  template <size_t Idx, class IOObj, class Tup>
620  std::enable_if_t<(Idx < std::tuple_size<Tup>::value)> serialize_tuple_element(IOObj& obj, const Tup& tup)
621  {
622  //serialize one element, then recurse

627  //recursive tuple deserialization for each tuple element
628  //read one tuple element after the other from the IOObject
629  //argument pack rs contains the elements that have been read already
630  template <class IOObj, class Tup, class... Ts>
631  std::enable_if_t<(sizeof...(Ts) == std::tuple_size<Tup>::value), IOResult<Tup>>
632  deserialize_tuple_element(IOObj& o, Tag<Tup>, const IOResult<Ts>&... rs)
633  {
634  //end of recursion
635  //number of arguments in rs is the same as the size of the tuple
636  //no more elements to read, so finalize the object
637  return epi::apply(
638  o,
639  [](const Ts&... ts) {
640  return Tup(ts...);
641  },
642  rs...);
643  }
644  template <class IOObj, class Tup, class... Ts>
645  std::enable_if_t<(sizeof...(Ts) < std::tuple_size<Tup>::value), IOResult<Tup>>
646  deserialize_tuple_element(IOObj& obj, Tag<Tup> tag, const IOResult<Ts>&... rs)
647  {
648  //get the next element of the tuple from the IO object
649  const size_t Idx = sizeof...(Ts);
650  auto r = obj.expect_element(make_tuple_element_name<Idx>(), Tag<std::tuple_element_t<Idx, Tup>>{});
651  //recurse, append the new element to the pack of arguments
652  return deserialize_tuple_element(obj, tag, rs..., r);
653  }
