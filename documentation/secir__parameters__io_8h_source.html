42 template <class Model>
43 IOResult<void> write_graph(const Graph<Model, MigrationParameters>& graph, const std::string& directory)
44 {
45  assert(graph.nodes().size() > 0 && "Graph Nodes are empty");
46 
47  std::string abs_path;
48  BOOST_OUTCOME_TRY(created, create_directory(directory, abs_path));
49 
50  if (created) {
51  log_info("Results are stored in {:s}/results.", epi::get_current_dir_name());
52  }
53  else {
54  log_info("Results are stored in {:s}/results. Files from previous "
55  "graph will be "
56  "overwritten",
57  epi::get_current_dir_name());
58  }
59 
60  //write two files per node
61  //one file that contains outgoing edges from the node
62  //one file for the model (parameters and population)
63  for (auto inode = size_t(0); inode < graph.nodes().size(); ++inode)
64  {
65  //node
66  auto& node = graph.nodes()[inode];
67  BOOST_OUTCOME_TRY(js_node_model, serialize_json(node.property));
68  Json::Value js_node(Json::objectValue);
69  js_node["NodeId"] = node.id;
70  js_node["Model"] = js_node_model;
71  auto node_filename = path_join(abs_path, "GraphNode" + std::to_string(inode) + ".json");
72  BOOST_OUTCOME_TRY(write_json(node_filename, js_node));
73 
74  //list of edges
75  Json::Value js_edges(Json::arrayValue);
76  for (auto& e : graph.out_edges(inode))
77  {
78  BOOST_OUTCOME_TRY(js_edge_params, serialize_json(e.property));
79  Json::Value js_edge{Json::objectValue};
80  js_edge["StartNodeIndex"] = Json::UInt64(e.start_node_idx);
81  js_edge["EndNodeIndex"] = Json::UInt64(e.end_node_idx);
82  js_edge["Parameters"] = js_edge_params;
83  js_edges.append(std::move(js_edge));
84  }
85  auto edge_filename = path_join(abs_path, "GraphEdges_node" + std::to_string(inode) + ".json");
86  BOOST_OUTCOME_TRY(write_json(edge_filename, js_edges));
87  }
88 
89  return success();
90 }

98 template <class Model>
99 IOResult<Graph<Model, MigrationParameters>> read_graph(const std::string& directory)
100 {
101  std::string abs_path;
102  if (!file_exists(directory, abs_path)) {
103  log_error("Directory {} does not exist.", directory);
104  return failure(StatusCode::FileNotFound, directory);
105  }
106 
107  auto graph = Graph<Model, MigrationParameters>{};
108 
109  //read nodes, as many as files are available
110  for (auto inode = 0; ; ++inode) {
111  auto node_filename = path_join(abs_path, "GraphNode" + std::to_string(inode) + ".json");
112  if (!file_exists(node_filename, node_filename)) {
113  break;
114  }
115  BOOST_OUTCOME_TRY(js_node, read_json(node_filename));
116  if (!js_node["NodeId"].isInt())
117  {
118  log_error("NodeId field must be an integer.");
119  return failure(StatusCode::InvalidType, node_filename + ", NodeId must be an integer.");
120  }
121  auto node_id = js_node["NodeId"].asInt();
122  BOOST_OUTCOME_TRY(model, deserialize_json(js_node["Model"], Tag<Model>{}));
123  graph.add_node(node_id, model);
124  }
125 
126  //read edges; nodes must already be available for that)
127  for (auto inode = size_t(0); inode < graph.nodes().size(); ++inode)
128  {
129  //list of edges
130  auto edge_filename = path_join(abs_path, "GraphEdges_node" + std::to_string(inode) + ".json");
131  BOOST_OUTCOME_TRY(js_edges, read_json(edge_filename));
132 
133  for (auto& e : js_edges) {
134  auto start_node_idx = inode;
135  auto js_end_node_idx = e["EndNodeIndex"];
136  if (!js_end_node_idx.isUInt64()) {
137  log_error("EndNodeIndex must be an integer.");
138  return failure(StatusCode::InvalidType, edge_filename + ", EndNodeIndex must be an integer.");
139  }
140  auto end_node_idx = js_end_node_idx.asUInt64();
141  if (end_node_idx >= graph.nodes().size()) {
142  log_error("EndNodeIndex not in range of number of graph nodes.");
143  return failure(StatusCode::OutOfRange, edge_filename + ", EndNodeIndex not in range of number of graph nodes.");
144  }
145  BOOST_OUTCOME_TRY(parameters, deserialize_json(e["Parameters"], Tag<MigrationParameters>{}));
146  graph.add_edge(start_node_idx, end_node_idx, parameters);
147  }
148  }
149 
150  return success(graph);
151 }

160 template <class Simulation>
161 IOResult<void> write_single_run_result(const int run, const epi::Graph<epi::SimulationNode<Simulation>, epi::MigrationEdge>& graph)
162 {
163  assert(graph.nodes().size() > 0 && "Graph Nodes are empty");
164 
165  std::string abs_path;
166  BOOST_OUTCOME_TRY(created, create_directory("results", abs_path));
167 
168  if (created) {
169  log_info("Results are stored in {:s}/results.", epi::get_current_dir_name());
170  }
171  else if (run == 0) {
172  log_info(
173  "Directory '{:s}' already exists. Results are stored in {:s}/results. Files from previous runs will be "
174  "overwritten",
175  epi::get_current_dir_name());
176  }
177 
178  //write sampled parameters for this run
179  //omit edges to save space as they are not sampled
180  for (auto inode = size_t(0); inode < graph.nodes().size(); ++inode) {
181  auto& node = graph.nodes()[inode];
182  BOOST_OUTCOME_TRY(js_node_model, serialize_json(node.property.get_result(), epi::IOF_OmitDistributions));
183  Json::Value js_node(Json::objectValue);
184  js_node["NodeId"] = node.id;
185  js_node["Model"] = js_node_model;
186  auto node_filename =
187  path_join(abs_path, "Parameters_run" + std::to_string(run) + "_node" + std::to_string(inode) + ".json");
188  BOOST_OUTCOME_TRY(write_json(node_filename, js_node));
189  }
190 
191  //write results for this run
192  std::vector<TimeSeries<double>> all_results;
193  std::vector<int> ids;
194 
195  ids.reserve(graph.nodes().size());
196  all_results.reserve(graph.nodes().size());
197  std::transform(graph.nodes().begin(), graph.nodes().end(), std::back_inserter(all_results), [](auto& node) {
198  return node.property.get_result();
199  });
200  std::transform(graph.nodes().begin(), graph.nodes().end(), std::back_inserter(ids), [](auto& node) {
201  return node.id;
202  });
203  BOOST_OUTCOME_TRY(
204  save_result(all_results, ids,
205  path_join(abs_path, ("Results_run" + std::to_string(run) + ".h5"))));
206 
207  return success();
208 }

321 template <class Model>
322 IOResult<void> extrapolate_rki_results(std::vector<Model>& model, const std::string& data_dir,
323  const std::string& results_dir, std::vector<int> const& region, Date date,
324  const std::vector<double>& scaling_factor_inf, double scaling_factor_icu,
325  int num_days)
326 {
327 
328  std::string id_name = "ID_County";
329  std::vector<double> age_ranges = {5., 10., 20., 25., 20., 20.};
330  assert(scaling_factor_inf.size() == age_ranges.size());
331 
332  std::vector<std::vector<int>> t_car_to_rec{model.size()}; // R9
333  std::vector<std::vector<int>> t_car_to_inf{model.size()}; // R3
334  std::vector<std::vector<int>> t_exp_to_car{model.size()}; // R2
335  std::vector<std::vector<int>> t_inf_to_rec{model.size()}; // R4
336  std::vector<std::vector<int>> t_inf_to_hosp{model.size()}; // R6
337  std::vector<std::vector<int>> t_hosp_to_rec{model.size()}; // R5
338  std::vector<std::vector<int>> t_hosp_to_icu{model.size()}; // R7
339  std::vector<std::vector<int>> t_icu_to_dead{model.size()}; // R10
340 
341  std::vector<std::vector<double>> mu_C_R{model.size()};
342  std::vector<std::vector<double>> mu_I_H{model.size()};
343  std::vector<std::vector<double>> mu_H_U{model.size()};
344 
345  std::vector<double> sum_mu_I_U(region.size(), 0);
346  std::vector<std::vector<double>> mu_I_U{model.size()};
347 
348  for (size_t county = 0; county < model.size(); county++) {
349  for (size_t group = 0; group < age_ranges.size(); group++) {
350 
351  t_car_to_inf[county].push_back(
352  static_cast<int>(2 * (model[county].parameters.template get<IncubationTime>()[AgeGroup(group)] -
353  model[county].parameters.template get<SerialInterval>()[AgeGroup(group)])));
354  t_car_to_rec[county].push_back(static_cast<int>(
355  t_car_to_inf[county][group] + 0.5 * model[county].parameters.template get<InfectiousTimeMild>()[AgeGroup(group)]));
356  t_exp_to_car[county].push_back(
357  static_cast<int>(2 * model[county].parameters.template get<SerialInterval>()[AgeGroup(group)] -
358  model[county].parameters.template get<IncubationTime>()[AgeGroup(group)]));
359  t_inf_to_rec[county].push_back(
360  static_cast<int>(model[county].parameters.template get<InfectiousTimeMild>()[AgeGroup(group)]));
361  t_inf_to_hosp[county].push_back(
362  static_cast<int>(model[county].parameters.template get<HomeToHospitalizedTime>()[AgeGroup(group)]));
363  t_hosp_to_rec[county].push_back(
364  static_cast<int>(model[county].parameters.template get<HospitalizedToHomeTime>()[AgeGroup(group)]));
365  t_hosp_to_icu[county].push_back(
366  static_cast<int>(model[county].parameters.template get<HospitalizedToICUTime>()[AgeGroup(group)]));
367  t_icu_to_dead[county].push_back(static_cast<int>(model[county].parameters.template get<ICUToDeathTime>()[AgeGroup(group)]));
368 
369  mu_C_R[county].push_back(model[county].parameters.template get<AsymptoticCasesPerInfectious>()[AgeGroup(group)]);
370  mu_I_H[county].push_back(model[county].parameters.template get<HospitalizedCasesPerInfectious>()[AgeGroup(group)]);
371  mu_H_U[county].push_back(model[county].parameters.template get<ICUCasesPerHospitalized>()[AgeGroup(group)]);
372 
373  sum_mu_I_U[county] += model[county].parameters.template get<ICUCasesPerHospitalized>()[AgeGroup(group)] *
374  model[county].parameters.template get<HospitalizedCasesPerInfectious>()[AgeGroup(group)];
375  mu_I_U[county].push_back(model[county].parameters.template get<ICUCasesPerHospitalized>()[AgeGroup(group)] *
376  model[county].parameters.template get<HospitalizedCasesPerInfectious>()[AgeGroup(group)]);
377  }
378  }
379 
380  std::vector<TimeSeries<double>> rki_data(
381  region.size(), TimeSeries<double>::zero(num_days, (size_t)InfectionState::Count * age_ranges.size()));
382 
383  for (size_t j = 0; j < static_cast<size_t>(num_days); j++) {
384  std::vector<std::vector<double>> num_inf(model.size(), std::vector<double>(age_ranges.size(), 0.0));
385  std::vector<std::vector<double>> num_death(model.size(), std::vector<double>(age_ranges.size(), 0.0));
386  std::vector<std::vector<double>> num_rec(model.size(), std::vector<double>(age_ranges.size(), 0.0));
387  std::vector<std::vector<double>> num_exp(model.size(), std::vector<double>(age_ranges.size(), 0.0));
388  std::vector<std::vector<double>> num_car(model.size(), std::vector<double>(age_ranges.size(), 0.0));
389  std::vector<std::vector<double>> num_hosp(model.size(), std::vector<double>(age_ranges.size(), 0.0));
390  std::vector<std::vector<double>> dummy_icu(model.size(), std::vector<double>(age_ranges.size(), 0.0));
391  std::vector<double> num_icu(model.size(), 0.0);
392 
393  BOOST_OUTCOME_TRY(details::read_rki_data(
394  path_join(data_dir, "all_county_age_ma_rki.json"), id_name, region, date, num_exp, num_car, num_inf, num_hosp,
395  dummy_icu, num_death, num_rec, t_car_to_rec, t_car_to_inf, t_exp_to_car, t_inf_to_rec, t_inf_to_hosp,
396  t_hosp_to_rec, t_hosp_to_icu, t_icu_to_dead, mu_C_R, mu_I_H, mu_H_U, scaling_factor_inf));
397  BOOST_OUTCOME_TRY(details::read_divi_data(path_join(data_dir, "county_divi.json"), id_name, region, date, num_icu));
398  BOOST_OUTCOME_TRY(num_population,
399  details::read_population_data(path_join(data_dir, "county_current_population.json"), id_name, region));
400 
401  for (size_t i = 0; i < region.size(); i++) {
402  for (size_t age = 0; age < age_ranges.size(); age++) {
403  rki_data[i][j]((size_t)InfectionState::Exposed + (size_t)epi::InfectionState::Count * age) =
404  num_exp[i][age];
405  rki_data[i][j]((size_t)InfectionState::Carrier + (size_t)InfectionState::Count * age) = num_car[i][age];
406  rki_data[i][j]((size_t)InfectionState::Infected + (size_t)InfectionState::Count * age) =
407  num_inf[i][age];
408  rki_data[i][j]((size_t)InfectionState::Hospitalized + (size_t)InfectionState::Count * age) =
409  num_hosp[i][age];
410  rki_data[i][j]((size_t)InfectionState::ICU + (size_t)InfectionState::Count * age) =
411  scaling_factor_icu * num_icu[i] * mu_I_U[i][age] / sum_mu_I_U[i];
412  rki_data[i][j]((size_t)InfectionState::Recovered + (size_t)InfectionState::Count * age) =
413  num_rec[i][age];
414  rki_data[i][j]((size_t)InfectionState::Dead + (size_t)InfectionState::Count * age) = num_death[i][age];
415  rki_data[i][j]((size_t)InfectionState::Susceptible + (size_t)InfectionState::Count * age) =
416  num_population[i][age] - num_exp[i][age] - num_car[i][age] - num_inf[i][age] - num_hosp[i][age] -
417  num_rec[i][age] - num_death[i][age] -
418  rki_data[i][j]((size_t)InfectionState::ICU + (size_t)InfectionState::Count * age);
419  }
420  }
421  date = offset_date_by_days(date, 1);
422  }
423  BOOST_OUTCOME_TRY(save_result(rki_data, region, path_join(results_dir, "Results_rki.h5")));
424 
425  auto rki_data_sum = epi::sum_nodes(std::vector<std::vector<TimeSeries<double>>>{rki_data});
426  BOOST_OUTCOME_TRY(save_result({rki_data_sum[0][0]}, {0}, path_join(results_dir, "Results_rki_sum.h5")));
427 
428  return success();
429 }
