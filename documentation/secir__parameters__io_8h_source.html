48 template <class Model>
49 IOResult<void> write_graph(const Graph<Model, MigrationParameters>& graph, const std::string& directory, int ioflags = IOF_None)
50 {
51  assert(graph.nodes().size() > 0 && "Graph Nodes are empty");
52 
53  std::string abs_path;
54  BOOST_OUTCOME_TRY(created, create_directory(directory, abs_path));
55 
56  if (created) {
57  log_info("Results are stored in {:s}/results.", mio::get_current_dir_name());
58  }
59  else {
60  log_info("Results are stored in {:s}/results. Files from previous "
61  "graph will be "
62  "overwritten",
63  mio::get_current_dir_name());
64  }
65 
66  //write two files per node
67  //one file that contains outgoing edges from the node
68  //one file for the model (parameters and population)
69  for (auto inode = size_t(0); inode < graph.nodes().size(); ++inode)
70  {
71  //node
72  auto& node = graph.nodes()[inode];
73  BOOST_OUTCOME_TRY(js_node_model, serialize_json(node.property, ioflags));
74  Json::Value js_node(Json::objectValue);
75  js_node["NodeId"] = node.id;
76  js_node["Model"] = js_node_model;
77  auto node_filename = path_join(abs_path, "GraphNode" + std::to_string(inode) + ".json");
78  BOOST_OUTCOME_TRY(write_json(node_filename, js_node));
79 
80  //list of edges
81  auto out_edges = graph.out_edges(inode);
82  if (out_edges.size()) {
83  Json::Value js_edges(Json::arrayValue);
84  for (auto& e : graph.out_edges(inode)) {
85  BOOST_OUTCOME_TRY(js_edge_params, serialize_json(e.property, ioflags));
86  Json::Value js_edge{Json::objectValue};
87  js_edge["StartNodeIndex"] = Json::UInt64(e.start_node_idx);
88  js_edge["EndNodeIndex"] = Json::UInt64(e.end_node_idx);
89  js_edge["Parameters"] = js_edge_params;
90  js_edges.append(std::move(js_edge));
91  }
92  auto edge_filename = path_join(abs_path, "GraphEdges_node" + std::to_string(inode) + ".json");
93  BOOST_OUTCOME_TRY(write_json(edge_filename, js_edges));
94  }
95  }
96 
97  return success();
98 }

106 template <class Model>
107 IOResult<Graph<Model, MigrationParameters>> read_graph(const std::string& directory, int ioflags = IOF_None)
108 {
109  std::string abs_path;
110  if (!file_exists(directory, abs_path)) {
111  log_error("Directory {} does not exist.", directory);
112  return failure(StatusCode::FileNotFound, directory);
113  }
114 
115  auto graph = Graph<Model, MigrationParameters>{};
116 
117  //read nodes, as many as files are available
118  for (auto inode = 0; ; ++inode) {
119  auto node_filename = path_join(abs_path, "GraphNode" + std::to_string(inode) + ".json");
120  if (!file_exists(node_filename, node_filename)) {
121  break;
122  }
123  BOOST_OUTCOME_TRY(js_node, read_json(node_filename));
124  if (!js_node["NodeId"].isInt())
125  {
126  log_error("NodeId field must be an integer.");
127  return failure(StatusCode::InvalidType, node_filename + ", NodeId must be an integer.");
128  }
129  auto node_id = js_node["NodeId"].asInt();
130  BOOST_OUTCOME_TRY(model, deserialize_json(js_node["Model"], Tag<Model>{}, ioflags));
131  graph.add_node(node_id, model);
132  }
133 
134  //read edges; nodes must already be available for that)
135  for (auto inode = size_t(0); inode < graph.nodes().size(); ++inode)
136  {
137  //list of edges
138  auto edge_filename = path_join(abs_path, "GraphEdges_node" + std::to_string(inode) + ".json");
139  BOOST_OUTCOME_TRY(js_edges, read_json(edge_filename));
140 
141  for (auto& e : js_edges) {
142  auto start_node_idx = inode;
143  auto js_end_node_idx = e["EndNodeIndex"];
144  if (!js_end_node_idx.isUInt64()) {
145  log_error("EndNodeIndex must be an integer.");
146  return failure(StatusCode::InvalidType, edge_filename + ", EndNodeIndex must be an integer.");
147  }
148  auto end_node_idx = js_end_node_idx.asUInt64();
149  if (end_node_idx >= graph.nodes().size()) {
150  log_error("EndNodeIndex not in range of number of graph nodes.");
151  return failure(StatusCode::OutOfRange, edge_filename + ", EndNodeIndex not in range of number of graph nodes.");
152  }
153  BOOST_OUTCOME_TRY(parameters, deserialize_json(e["Parameters"], Tag<MigrationParameters>{}, ioflags));
154  graph.add_edge(start_node_idx, end_node_idx, parameters);
155  }
156  }
157 
158  return success(graph);
159 }

274 template <class Model>
275 IOResult<void> extrapolate_rki_results(std::vector<Model>& model, const std::string& data_dir,
276  const std::string& results_dir, std::vector<int> const& region, Date date,
277  const std::vector<double>& scaling_factor_inf, double scaling_factor_icu,
278  int num_days)
279 {
280 
281  std::string id_name = "ID_County";
282  std::vector<double> age_ranges = {5., 10., 20., 25., 20., 20.};
283  assert(scaling_factor_inf.size() == age_ranges.size());
284 
285  std::vector<std::vector<int>> t_car_to_rec{model.size()}; // R9
286  std::vector<std::vector<int>> t_car_to_inf{model.size()}; // R3
287  std::vector<std::vector<int>> t_exp_to_car{model.size()}; // R2
288  std::vector<std::vector<int>> t_inf_to_rec{model.size()}; // R4
289  std::vector<std::vector<int>> t_inf_to_hosp{model.size()}; // R6
290  std::vector<std::vector<int>> t_hosp_to_rec{model.size()}; // R5
291  std::vector<std::vector<int>> t_hosp_to_icu{model.size()}; // R7
292  std::vector<std::vector<int>> t_icu_to_dead{model.size()}; // R10
293 
294  std::vector<std::vector<double>> mu_C_R{model.size()};
295  std::vector<std::vector<double>> mu_I_H{model.size()};
296  std::vector<std::vector<double>> mu_H_U{model.size()};
297 
298  std::vector<double> sum_mu_I_U(region.size(), 0);
299  std::vector<std::vector<double>> mu_I_U{model.size()};
300 
301  for (size_t county = 0; county < model.size(); county++) {
302  for (size_t group = 0; group < age_ranges.size(); group++) {
303 
304  t_car_to_inf[county].push_back(
305  static_cast<int>(2 * (model[county].parameters.template get<IncubationTime>()[AgeGroup(group)] -
306  model[county].parameters.template get<SerialInterval>()[AgeGroup(group)])));
307  t_car_to_rec[county].push_back(static_cast<int>(
308  t_car_to_inf[county][group] + 0.5 * model[county].parameters.template get<InfectiousTimeMild>()[AgeGroup(group)]));
309  t_exp_to_car[county].push_back(
310  static_cast<int>(2 * model[county].parameters.template get<SerialInterval>()[AgeGroup(group)] -
311  model[county].parameters.template get<IncubationTime>()[AgeGroup(group)]));
312  t_inf_to_rec[county].push_back(
313  static_cast<int>(model[county].parameters.template get<InfectiousTimeMild>()[AgeGroup(group)]));
314  t_inf_to_hosp[county].push_back(
315  static_cast<int>(model[county].parameters.template get<HomeToHospitalizedTime>()[AgeGroup(group)]));
316  t_hosp_to_rec[county].push_back(
317  static_cast<int>(model[county].parameters.template get<HospitalizedToHomeTime>()[AgeGroup(group)]));
318  t_hosp_to_icu[county].push_back(
319  static_cast<int>(model[county].parameters.template get<HospitalizedToICUTime>()[AgeGroup(group)]));
320  t_icu_to_dead[county].push_back(static_cast<int>(model[county].parameters.template get<ICUToDeathTime>()[AgeGroup(group)]));
321 
322  mu_C_R[county].push_back(model[county].parameters.template get<AsymptomaticCasesPerInfectious>()[AgeGroup(group)]);
323  mu_I_H[county].push_back(model[county].parameters.template get<HospitalizedCasesPerInfectious>()[AgeGroup(group)]);
324  mu_H_U[county].push_back(model[county].parameters.template get<ICUCasesPerHospitalized>()[AgeGroup(group)]);
325 
326  sum_mu_I_U[county] += model[county].parameters.template get<ICUCasesPerHospitalized>()[AgeGroup(group)] *
327  model[county].parameters.template get<HospitalizedCasesPerInfectious>()[AgeGroup(group)];
328  mu_I_U[county].push_back(model[county].parameters.template get<ICUCasesPerHospitalized>()[AgeGroup(group)] *
329  model[county].parameters.template get<HospitalizedCasesPerInfectious>()[AgeGroup(group)]);
330  }
331  }
332 
333  std::vector<TimeSeries<double>> rki_data(
334  region.size(), TimeSeries<double>::zero(num_days, (size_t)InfectionState::Count * age_ranges.size()));
335 
336  for (size_t j = 0; j < static_cast<size_t>(num_days); j++) {
337  std::vector<std::vector<double>> num_inf(model.size(), std::vector<double>(age_ranges.size(), 0.0));
338  std::vector<std::vector<double>> num_death(model.size(), std::vector<double>(age_ranges.size(), 0.0));
339  std::vector<std::vector<double>> num_rec(model.size(), std::vector<double>(age_ranges.size(), 0.0));
340  std::vector<std::vector<double>> num_exp(model.size(), std::vector<double>(age_ranges.size(), 0.0));
341  std::vector<std::vector<double>> num_car(model.size(), std::vector<double>(age_ranges.size(), 0.0));
342  std::vector<std::vector<double>> num_hosp(model.size(), std::vector<double>(age_ranges.size(), 0.0));
343  std::vector<std::vector<double>> dummy_icu(model.size(), std::vector<double>(age_ranges.size(), 0.0));
344  std::vector<double> num_icu(model.size(), 0.0);
345 
346  BOOST_OUTCOME_TRY(details::read_rki_data(
347  path_join(data_dir, "cases_all_county_age_ma7.json"), id_name, region, date, num_exp, num_car, num_inf, num_hosp,
348  dummy_icu, num_death, num_rec, t_car_to_rec, t_car_to_inf, t_exp_to_car, t_inf_to_rec, t_inf_to_hosp,
349  t_hosp_to_rec, t_hosp_to_icu, t_icu_to_dead, mu_C_R, mu_I_H, mu_H_U, scaling_factor_inf));
350  BOOST_OUTCOME_TRY(details::read_divi_data(path_join(data_dir, "county_divi.json"), id_name, region, date, num_icu));
351  BOOST_OUTCOME_TRY(num_population,
352  details::read_population_data(path_join(data_dir, "county_current_population.json"), id_name, region));
353 
354  for (size_t i = 0; i < region.size(); i++) {
355  for (size_t age = 0; age < age_ranges.size(); age++) {
356  rki_data[i][j]((size_t)InfectionState::Exposed + (size_t)mio::InfectionState::Count * age) =
357  num_exp[i][age];
358  rki_data[i][j]((size_t)InfectionState::Carrier + (size_t)InfectionState::Count * age) = num_car[i][age];
359  rki_data[i][j]((size_t)InfectionState::Infected + (size_t)InfectionState::Count * age) =
360  num_inf[i][age];
361  rki_data[i][j]((size_t)InfectionState::Hospitalized + (size_t)InfectionState::Count * age) =
362  num_hosp[i][age];
363  rki_data[i][j]((size_t)InfectionState::ICU + (size_t)InfectionState::Count * age) =
364  scaling_factor_icu * num_icu[i] * mu_I_U[i][age] / sum_mu_I_U[i];
365  rki_data[i][j]((size_t)InfectionState::Recovered + (size_t)InfectionState::Count * age) =
366  num_rec[i][age];
367  rki_data[i][j]((size_t)InfectionState::Dead + (size_t)InfectionState::Count * age) = num_death[i][age];
368  rki_data[i][j]((size_t)InfectionState::Susceptible + (size_t)InfectionState::Count * age) =
369  num_population[i][age] - num_exp[i][age] - num_car[i][age] - num_inf[i][age] - num_hosp[i][age] -
370  num_rec[i][age] - num_death[i][age] -
371  rki_data[i][j]((size_t)InfectionState::ICU + (size_t)InfectionState::Count * age);
372  }
373  }
374  date = offset_date_by_days(date, 1);
375  }
376  BOOST_OUTCOME_TRY(save_result(rki_data, region, path_join(results_dir, "Results_rki.h5")));
377 
378  auto rki_data_sum = mio::sum_nodes(std::vector<std::vector<TimeSeries<double>>>{rki_data});
379  BOOST_OUTCOME_TRY(save_result({rki_data_sum[0][0]}, {0}, path_join(results_dir, "Results_rki_sum.h5")));
380 
381  return success();
382 }
