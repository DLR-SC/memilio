159  //for X = template<T1, T2> X => BindTail<X, A>::type<B> = X<B, A>
160  template<template<class...> class F, class... Tail>
161  struct BindTail
162  {
163  template<class... Head>
164  struct type : F<Head..., Tail...> {};
