486  template <class Iter>
487  static void inclusive_exclusive_sum_rec(Iter b, Iter e, Matrix& sum)
488  {
489  if (b != e) {
490  sum = sum + std::get<Matrix>(*b) - (sum.array() * std::get<Matrix>(*b).array()).matrix();
491  inclusive_exclusive_sum_rec(++b, e, sum);
492  }
493  }
494  template <class Tuple>
495  static Matrix inclusive_exclusive_sum(const std::vector<Tuple>& v)
496  {
497  assert(!v.empty());
498  auto& m = std::get<Matrix>(v.front());
499  auto sum = m.eval();
500  inclusive_exclusive_sum_rec(v.begin() + 1, v.end(), sum);
501  return sum;
502  }

510 template <class D>
511 void Dampings<D>::finalize() const
512 {
513  using std::get;
514 
515  if (m_accumulated_dampings_cached.empty()) {
516  m_accumulated_dampings_cached.emplace_back(Matrix::Zero(m_shape.rows(), m_shape.cols()),
517  SimulationTime(std::numeric_limits<double>::lowest()));
518 
519  std::vector<std::tuple<std::reference_wrapper<const Matrix>, DampingLevel, DampingType>> active_by_type;
520  std::vector<std::tuple<Matrix, DampingLevel>> sum_by_level;
521  for (auto& damping : m_dampings) {
522  update_active_dampings(damping, active_by_type, sum_by_level);
523  auto combined_damping = inclusive_exclusive_sum(sum_by_level);
524  assert((combined_damping.array() <= 1).all() && (combined_damping.array() >= 0).all() &&
525  "unexpected error, accumulated damping out of range.");
526  if (floating_point_equal(double(get<SimulationTime>(damping)),
527  double(get<SimulationTime>(m_accumulated_dampings_cached.back())), 1e-15, 1e-15)) {
528  std::get<Matrix>(m_accumulated_dampings_cached.back()) = combined_damping;
529  }
530  else {
531  m_accumulated_dampings_cached.emplace_back(combined_damping, get<SimulationTime>(damping));
532  }
533  }
534 
535  m_accumulated_dampings_cached.emplace_back(get<Matrix>(m_accumulated_dampings_cached.back()),
536  SimulationTime(std::numeric_limits<double>::max()));
537  }
538 }

541 void Dampings<D>::add_(const value_type& damping)
542 {
543  assert(damping.get_shape() == m_shape && "Inconsistent matrix shape.");
544  insert_sorted_replace(m_dampings, damping, [](auto& tup1, auto& tup2) {
545  return std::make_tuple(tup1.get_time(), int(tup1.get_type()), int(tup1.get_level())) <
546  std::make_tuple(tup2.get_time(), int(tup2.get_type()), int(tup2.get_level()));
547  });
548  m_accumulated_dampings_cached.clear();

551 template<class S>
552 void Dampings<S>::update_active_dampings(
553  const value_type& damping,
554  std::vector<std::tuple<std::reference_wrapper<const Matrix>, DampingLevel, DampingType>>& active_by_type,
555  std::vector<std::tuple<Matrix, DampingLevel>>& sum_by_level)
556 {
557  using std::get;
558 
559  const int MatrixIdx = 0;
560 
561  auto iter_active_same_type = std::find_if(active_by_type.begin(), active_by_type.end(), [&damping](auto& active) {
562  return get<DampingLevel>(active) == get<DampingLevel>(damping) &&
563  get<DampingType>(active) == get<DampingType>(damping);
564  });
565  if (iter_active_same_type != active_by_type.end()) {
566  //replace active of the same type and level
567  auto& active_same_type = *iter_active_same_type;
568  auto& sum_same_level = *std::find_if(sum_by_level.begin(), sum_by_level.end(), [&damping](auto& sum) {
569  return get<DampingLevel>(sum) == get<DampingLevel>(damping);
570  });
571  get<MatrixIdx>(sum_same_level) += get<MatrixIdx>(damping) - get<MatrixIdx>(active_same_type).get();
572  get<MatrixIdx>(active_same_type) = get<MatrixIdx>(damping);
573  }
574  else {
575  //add new type
576  active_by_type.emplace_back(get<MatrixIdx>(damping), get<DampingLevel>(damping), get<DampingType>(damping));
577 
578  auto iter_sum_same_level = std::find_if(sum_by_level.begin(), sum_by_level.end(), [&damping](auto& sum) {
579  return get<DampingLevel>(sum) == get<DampingLevel>(damping);
580  });
581  if (iter_sum_same_level != sum_by_level.end()) {
582  //add to existing level
583  get<MatrixIdx>(*iter_sum_same_level) += get<MatrixIdx>(damping);
584  }
585  else {
586  //add new level
587  sum_by_level.emplace_back(get<MatrixIdx>(damping), get<DampingLevel>(damping));
588  }
589  }
590 }
