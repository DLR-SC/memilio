387  inline Slice::difference_type outer_offset(difference_type const& inner) const {
388 
389  // calculate the outer offset from the inner offset
390 
391  // first unravel the inner index into an index (i,j,k) for a 3-dim array with dims (dl, idx_sequence.n, dr)
392  auto dv = std::div(inner, seq.n*dr);
393  difference_type i = dv.quot;
394  dv = std::div(dv.rem, dr);
395  difference_type j = dv.quot*seq.stride + seq.start;
396  difference_type k = dv.rem;
397 
398  // then flatten the index for a 3-dim array with dims (dl, di, dr)
399  return i*di*dr + j*dr + k;
400  }
