86  template <size_t I, typename Index>
87  std::enable_if_t< (I < (Index::size - 1) ), std::pair<size_t, size_t>> flatten_index(Index const& indices, Index const& dimensions)
88  {
89  assert(mio::get<I>(indices) < mio::get<I>(dimensions));
90 
91  size_t val, prod;
92  std::tie(val, prod) = flatten_index<I+1>(indices, dimensions);
93

413  inline Slice::difference_type outer_offset(difference_type const& inner) const {
414 
415  // calculate the outer offset from the inner offset
416 
417  // first unravel the inner index into an index (i,j,k) for a 3-dim array with dims (dl, idx_sequence.n, dr)
418  auto dv = std::div(inner, seq.n*dr);
419  difference_type i = dv.quot;
420  dv = std::div(dv.rem, dr);
421  difference_type j = dv.quot*seq.stride + seq.start;
422  difference_type k = dv.rem;
423 
424  // then flatten the index for a 3-dim array with dims (dl, di, dr)
425  return i*di*dr + j*dr + k;
426  }
