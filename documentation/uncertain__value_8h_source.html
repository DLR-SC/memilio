158  static IOResult<UncertainValue> deserialize(IOContext& io)
159  {

161  if (!(io.flags() & IOF_OmitValues) && !(io.flags() & IOF_OmitDistributions)) {
162  auto v = obj.expect_element("Value", Tag<ScalarType>{});
163  auto d = obj.expect_optional("Distribution", Tag<std::shared_ptr<ParameterDistribution>>{});

171  return uv;
