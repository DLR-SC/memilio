691  void apply_constraints()
692  {
693  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
694  log_warning("Constraint check: Parameter Seasonality changed from {:0.4f} to {:d}",
695  this->get<Seasonality>(), 0);
696  this->set<Seasonality>(0);
697  }
698 
699  if (this->get<ICUCapacity>() < 0.0) {
700  log_warning("Constraint check: Parameter ICUCapacity changed from {:0.4f} to {:d}",
701  this->get<ICUCapacity>(), 0);
702  this->set<ICUCapacity>(0);
703  }
704 
705  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
706 
707  if (this->get<IncubationTime>()[i] < 2.0) {
708  log_warning("Constraint check: Parameter IncubationTime changed from {:.4f} to {:.4f}",
709  this->get<IncubationTime>()[i], 2.0);
710  this->get<IncubationTime>()[i] = 2.0;
711  }
712 
713  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
714  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}",
715  this->get<SerialInterval>()[i], 0.5 * this->get<IncubationTime>()[i] + 0.5);
716  this->get<SerialInterval>()[i] = 0.5 * this->get<IncubationTime>()[i] + 0.5;
717  }
718  else if (this->get<SerialInterval>()[i] > this->get<IncubationTime>()[i] - 0.5) {
719  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}",
720  this->get<SerialInterval>()[i], this->get<IncubationTime>()[i] - 0.5);
721  this->get<SerialInterval>()[i] = this->get<IncubationTime>()[i] - 0.5;
722  }
723 
724  if (this->get<InfectiousTimeMild>()[i] < 1.0) {
725  log_warning("Constraint check: Parameter InfectiousTimeMild changed from {:.4f} to {:.4f}",
726  this->get<InfectiousTimeMild>()[i], 1.0);
727  this->get<InfectiousTimeMild>()[i] = 1.0;
728  }
729 
730  if (this->get<HospitalizedToHomeTime>()[i] < 1.0) {
731  log_warning("Constraint check: Parameter HospitalizedToHomeTime changed from {:.4f} to {:.4f}",
732  this->get<HospitalizedToHomeTime>()[i], 1.0);
733  this->get<HospitalizedToHomeTime>()[i] = 1.0;
734  }
735 
736  if (this->get<HomeToHospitalizedTime>()[i] < 1.0) {
737  log_warning("Constraint check: Parameter HomeToHospitalizedTime changed from {:.4f} to {:.4f}",
738  this->get<HomeToHospitalizedTime>()[i], 1.0);
739  this->get<HomeToHospitalizedTime>()[i] = 1.0;
740  }
741 
742  if (this->get<HospitalizedToICUTime>()[i] < 1.0) {
743  log_warning("Constraint check: Parameter HospitalizedToICUTime changed from {:.4f} to {:.4f}",
744  this->get<HospitalizedToICUTime>()[i], 1.0);
745  this->get<HospitalizedToICUTime>()[i] = 1.0;
746  }
747 
748  if (this->get<ICUToHomeTime>()[i] < 1.0) {
749  log_warning("Constraint check: Parameter ICUToHomeTime changed from {:.4f} to {:.4f}",
750  this->get<ICUToHomeTime>()[i], 1.0);
751  this->get<ICUToHomeTime>()[i] = 1.0;
752  }
753 
754  if (this->get<ICUToDeathTime>()[i] < 1.0) {
755  log_warning("Constraint check: Parameter ICUToDeathTime changed from {:.4f} to {:.4f}",
756  this->get<ICUToDeathTime>()[i], 1.0);
757  this->get<ICUToDeathTime>()[i] = 1.0;
758  }

767  }
768 
769  if (this->get<InfectionProbabilityFromContact>()[i] < 0.0) {

772  this->get<InfectionProbabilityFromContact>()[i] = 0;
773  }
774 
775  if (this->get<RelativeCarrierInfectability>()[i] < 0.0) {

819  void check_constraints() const
820  {
821  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
822  log_warning("Constraint check: Parameter m_seasonality smaller {:d} or larger {:d}", 0, 0.5);
823  }
824 
825  if (this->get<ICUCapacity>() < 0.0) {
826  log_warning("Constraint check: Parameter m_icu_capacity smaller {:d}", 0);
827  }
828 
829  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
830 
831  if (this->get<IncubationTime>()[i] < 2.0) {
832  log_error("Constraint check: Parameter IncubationTime {:.4f} smaller {:.4f}",
833  this->get<IncubationTime>()[i], 2.0);
834  }
835 
836  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
837  log_error("Constraint check: Parameter SerialInterval {:.4f} smaller {:.4f}",
838  this->get<SerialInterval>()[i], 0.5 * this->get<IncubationTime>()[i] + 0.5);
839  }
840  else if (this->get<SerialInterval>()[i] > this->get<IncubationTime>()[i] - 0.5) {
841  log_error("Constraint check: Parameter SerialInterval {:.4f} smaller {:.4f}",
842  this->get<SerialInterval>()[i], this->get<IncubationTime>()[i] - 0.5);
843  }
844 
845  if (this->get<InfectiousTimeMild>()[i] < 1.0) {
846  log_error("Constraint check: Parameter InfectiousTimeMild {:.4f} smaller {:.4f}",
847  this->get<InfectiousTimeMild>()[i], 1.0);
848  }
