65 template <class T>
66 struct JsonType<T, std::enable_if_t<conjunction_v<is_small_integral<T>, std::is_signed<T>>>> : std::true_type {
67  static IOResult<T> transform(const Json::Value& js)
68  {
69  if (js.isInt()) {
70  int i = js.asInt();
71  if (i >= int(std::numeric_limits<T>::min()) && i <= int(std::numeric_limits<T>::max())) {
72  return success(T(i));
73  }

498 inline IOResult<void> write_json(const std::string& path, const Json::Value& js_value)
499 {
500  std::ofstream ofs(path);
501  if (ofs.is_open()) {
502  Json::StreamWriterBuilder swb;
503  auto js_writer = std::unique_ptr<Json::StreamWriter>(swb.newStreamWriter());
504  js_writer->write(js_value, &ofs);
505  if (ofs) {
506  return success();
507  }
508  else {
509  return failure(StatusCode::UnknownError, "Unknown error writing json.");
510  }
511  }
512  else {
513  return failure(StatusCode::FileNotFound, path);
514  }
515 }

555 inline IOResult<Json::Value> read_json(const std::string& path)
556 {
557  std::ifstream ifs(path);
558  if (ifs.is_open()) {
559  Json::CharReaderBuilder crb;
560  std::string err_msg;
561  Json::Value js_value;
562  if (Json::parseFromStream(crb, ifs, &js_value, &err_msg)) {
563  return success(js_value);
564  }
565  return failure(StatusCode::UnknownError, path + ", " + err_msg);
566  }
567  return failure(StatusCode::FileNotFound, path);
568 }
