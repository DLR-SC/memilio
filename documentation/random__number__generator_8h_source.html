306  template <class RNG>
307  result_type operator()(RNG& rng, param_type p)
308  {
309  auto weights = p.weights();
310  if (weights.size() <= 1) {
311  return 0;
312  }
313  auto sum = std::accumulate(weights.begin(), weights.end(), 0.0);
314  auto u =
315  std::uniform_real_distribution<double>()(rng, std::uniform_real_distribution<double>::param_type{0.0, sum});
316  auto intermediate_sum = 0.0;
317  for (size_t i = 0; i < weights.size(); ++i) {
318  intermediate_sum += weights.get_ptr()[i];
319  if (u < intermediate_sum) {
320  return i;
321  }
322  }
323  assert(false && "this should never happen.");
324  return result_type(-1);
325  }
326 
327 private:
328  param_type m_params;
