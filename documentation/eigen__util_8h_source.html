128 template <class M, std::enable_if_t<is_dynamic_matrix<M>::value, int> = 0>
129 auto slice(M&& m, Seq<Eigen::Index> rows, Seq<Eigen::Index> cols)
130 {
131  assert(rows.start + rows.stride * rows.n <= m.rows());
132  assert(cols.start + cols.stride * cols.n <= m.rows());
133 
134  auto majSpec = std::remove_reference_t<M>::IsRowMajor ? rows : cols;
135  auto minSpec = std::remove_reference_t<M>::IsRowMajor ? cols : rows;
136  auto majStride = majSpec.stride * minor_size(m);
137  auto minStride = minSpec.stride;
138  auto data = m.data() + majSpec.start * minor_size(m) + minSpec.start;
139 
140  return Eigen::Map<CVPlainMatrixT<std::remove_reference_t<M>>, Eigen::Unaligned, Eigen::Stride<Eigen::Dynamic, Eigen::Dynamic>>(
141  data, rows.n, cols.n, {majStride, minStride});
142 }

219 template <class M, bool IsConst = false>
220 class RowMajorIterator
221 {
222 public:
223  using MatrixRef = std::conditional_t<IsConst, const M&, M&>;
224  using MatrixPtr = std::conditional_t<IsConst, const M*, M*>;
225  static_assert(IsConst || details::IsElementReference<M>::value,
226  "Iterator must be const if matrix is not in memory.");
227 
228  using iterator_category = std::random_access_iterator_tag;
229  using value_type = typename M::Scalar;
230  using reference = std::conditional_t<IsConst, const value_type&, value_type&>;
231  using difference_type = Eigen::Index;
232 
233  // Operator-> returns a pointer to an element if the element can be referenced.
234  // If the matrix returns temporaries, a proxy is returned that holds a local copy
235  // and forwards the address of that copy. Adress is pointer-to-const because modifying
236  // the local inside the proxy is almost certainly not what is intended.
237  struct Proxy {
238  value_type value;
239  const value_type* operator->() const
240  {
241  return &value;
242  }
243  };
244  using pointer = std::conditional_t<details::IsElementReference<MatrixRef>::value,
245  std::conditional_t<IsConst, const value_type*, value_type*>, Proxy>;
