111  template <class IOContext>
112  void serialize(IOContext& io) const
113  {
114  auto obj = io.create_object("World");
115  obj.add_element("num_agegroups", parameters.get_num_groups());
116  std::vector<Trip> trips;
117  TripList trip_list = get_trip_list();
118  for (size_t i = 0; i < trip_list.num_trips(false); i++) {
119  trips.push_back(trip_list.get_next_trip(false));
120  trip_list.increase_index();
121  }
122  trip_list.reset_index();
123  for (size_t i = 0; i < trip_list.num_trips(true); i++) {
124  trips.push_back(trip_list.get_next_trip(true));
125  trip_list.increase_index();
126  }
127  obj.add_list("trips", trips.begin(), trips.end());
128  obj.add_list("locations", get_locations().begin(), get_locations().end());
129  obj.add_list("persons", get_persons().begin(), get_persons().end());
130  obj.add_element("use_migration_rules", m_use_migration_rules);
131  }

369  // move a person to another location. this requires that location is part of this world.
377  inline void migrate(PersonId person, LocationId destination, TransportMode mode = TransportMode::Unknown,
378  const std::vector<uint32_t>& cells = {0})
379  {
380  LocationId origin = get_location(person).get_id();
381  const bool has_moved = mio::abm::migrate(get_person(person), get_location(destination), mode, cells);
382  // if the person has moved, invalidate exposure caches but keep population caches valid
383  if (has_moved) {
384  m_are_exposure_caches_valid = false;
385  if (m_is_local_population_cache_valid) {
386  --m_local_population_cache[origin.get()];
387  ++m_local_population_cache[destination.get()];
388  }
389  }
390  }

398  inline void interact(PersonId person, TimePoint t, TimeSpan dt)
399  {
400  if (!m_are_exposure_caches_valid) {
401  // checking caches is only needed for external calls
402  // during simulation (i.e. in evolve()), the caches are computed in begin_step
403  compute_exposure_caches(t, dt);
404  m_are_exposure_caches_valid = true;
405  }
406  auto personal_rng = PersonalRandomNumberGenerator(m_rng, get_person(person));
407  mio::abm::interact(personal_rng, get_person(person), get_location(person),
408  m_air_exposure_rates_cache[get_location(person).get_id().get()],
409  m_contact_exposure_rates_cache[get_location(person).get_id().get()], t, dt, parameters);
410  }
