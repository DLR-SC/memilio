427  void apply_constraints()
428  {
429  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
430  log_warning("Constraint check: Parameter Seasonality changed from {:0.4f} to {:d}",
431  this->get<Seasonality>(), 0);
432  this->set<Seasonality>(0);
433  }
434 
435  if (this->get<ICUCapacity>() < 0.0) {
436  log_warning("Constraint check: Parameter ICUCapacity changed from {:0.4f} to {:d}",
437  this->get<ICUCapacity>(), 0);
438  this->set<ICUCapacity>(0);
439  }
440 
441  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
442 
443  if (this->get<IncubationTime>()[i] < 2.0) {
444  log_warning("Constraint check: Parameter IncubationTime changed from {:.4f} to {:.4f}",
445  this->get<IncubationTime>()[i], 2.0);
446  this->get<IncubationTime>()[i] = 2.0;
447  }
448 
449  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
450  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}",
451  this->get<SerialInterval>()[i], 0.5 * this->get<IncubationTime>()[i] + 0.5);
452  this->get<SerialInterval>()[i] = 0.5 * this->get<IncubationTime>()[i] + 0.5;
453  }
454  else if (this->get<SerialInterval>()[i] > this->get<IncubationTime>()[i] - 0.5) {
455  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}",
456  this->get<SerialInterval>()[i], this->get<IncubationTime>()[i] - 0.5);
457  this->get<SerialInterval>()[i] = this->get<IncubationTime>()[i] - 0.5;
458  }
459 
460  if (this->get<InfectiousTimeMild>()[i] < 1.0) {
461  log_warning("Constraint check: Parameter InfectiousTimeMild changed from {:.4f} to {:.4f}",
462  this->get<InfectiousTimeMild>()[i], 1.0);
463  this->get<InfectiousTimeMild>()[i] = 1.0;
464  }
465 
466  if (this->get<HospitalizedToHomeTime>()[i] < 1.0) {
467  log_warning("Constraint check: Parameter HospitalizedToHomeTime changed from {:.4f} to {:.4f}",

476  }

480  this->get<HospitalizedToICUTime>()[i], 1.0);
481  this->get<HospitalizedToICUTime>()[i] = 1.0;
482  }
483 
484  if (this->get<ICUToHomeTime>()[i] < 1.0) {
485  log_warning("Constraint check: Parameter ICUToHomeTime changed from {:.4f} to {:.4f}",
486  this->get<ICUToHomeTime>()[i], 1.0);
487  this->get<ICUToHomeTime>()[i] = 1.0;
488  }
489 
490  if (this->get<ICUToDeathTime>()[i] < 1.0) {
491  log_warning("Constraint check: Parameter ICUToDeathTime changed from {:.4f} to {:.4f}",
492  this->get<ICUToDeathTime>()[i], 1.0);
493  this->get<ICUToDeathTime>()[i] = 1.0;
494  }
495 
496  auto t_inf_asymp = 1.0 / (0.5 / (this->get<IncubationTime>()[i] - this->get<SerialInterval>()[i])) +
497  0.5 * this->get<InfectiousTimeMild>()[i];

503  }

508  this->get<InfectionProbabilityFromContact>()[i] = 0;

555  void check_constraints() const
556  {
557  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
558  log_warning("Constraint check: Parameter m_seasonality smaller {:d} or larger {:d}", 0, 0.5);
559  }
560 
561  if (this->get<ICUCapacity>() < 0.0) {
562  log_warning("Constraint check: Parameter m_icu_capacity smaller {:d}", 0);
563  }
564 
565  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
566 
567  if (this->get<IncubationTime>()[i] < 2.0) {
568  log_error("Constraint check: Parameter IncubationTime {:.4f} smaller {:.4f}",
569  this->get<IncubationTime>()[i], 2.0);
570  }
571 
572  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
573  log_error("Constraint check: Parameter SerialInterval {:.4f} smaller {:.4f}",
574  this->get<SerialInterval>()[i], 0.5 * this->get<IncubationTime>()[i] + 0.5);
575  }
576  else if (this->get<SerialInterval>()[i] > this->get<IncubationTime>()[i] - 0.5) {
577  log_error("Constraint check: Parameter SerialInterval {:.4f} smaller {:.4f}",
578  this->get<SerialInterval>()[i], this->get<IncubationTime>()[i] - 0.5);
579  }
580 
581  if (this->get<InfectiousTimeMild>()[i] < 1.0) {
582  log_error("Constraint check: Parameter InfectiousTimeMild {:.4f} smaller {:.4f}",
583  this->get<InfectiousTimeMild>()[i], 1.0);
584  }
585 
586  if (this->get<HospitalizedToHomeTime>()[i] < 1.0) {
587  log_error("Constraint check: Parameter HospitalizedToHomeTime {:.4f} smaller {:.4f}",
588  this->get<HospitalizedToHomeTime>()[i], 1.0);
589  }
590 
591  if (this->get<HomeToHospitalizedTime>()[i] < 1.0) {
592  log_error("Constraint check: Parameter HomeToHospitalizedTime {:.4f} smaller {:.4f}",
593  this->get<HomeToHospitalizedTime>()[i], 1.0);
594  }
595 
596  if (this->get<HospitalizedToICUTime>()[i] < 1.0) {
597  log_error("Constraint check: Parameter HospitalizedToICUTime {:.4f} smaller {:.4f}",
598  this->get<HospitalizedToICUTime>()[i], 1.0);
599  }
600 
601  if (this->get<ICUToHomeTime>()[i] < 1.0) {
602  log_error("Constraint check: Parameter ICUToHomeTime {:.4f} smaller {:.4f}",
603  this->get<ICUToHomeTime>()[i], 1.0);
604  }
605 
606  if (abs(this->get<InfectiousTimeAsymptomatic>()[i] -
607  1.0 / (0.5 / (this->get<IncubationTime>()[i] - this->get<SerialInterval>()[i])) -
608  0.5 * this->get<InfectiousTimeMild>()[i]) > 1e-12) {
609  log_warning("Constraint check: Parameter InfectiousTimeAsymptomatic not set as fully dependent on "
610  "tinc, tserint and tinfmild, as proposed by "
611  "https://www.medrxiv.org/content/10.1101/2020.04.04.20053637v1.");
612  }
613 
614  if (this->get<ICUToDeathTime>()[i] < 1.0) {
615  log_error("Constraint check: Parameter ICUToDeathTime {:.4f} smaller {:.4f}",
616  this->get<ICUToDeathTime>()[i], 1.0);
617  }
618 
619  if (this->get<InfectionProbabilityFromContact>()[i] < 0.0) {
620  log_warning("Constraint check: Parameter InfectionProbabilityFromContact smaller {:d}", 0);
621  }
622 
623  if (this->get<RelativeCarrierInfectability>()[i] < 0.0) {
624  log_warning("Constraint check: Parameter RelativeCarrierInfectability smaller {:d}", 0);
625  }
626 
627  if (this->get<AsymptomaticCasesPerInfectious>()[i] < 0.0 ||
628  this->get<AsymptomaticCasesPerInfectious>()[i] > 1.0) {
629  log_warning("Constraint check: Parameter AsymptomaticCasesPerInfectious smaller {:d} or larger {:d}", 0,
630  1);
631  }
632 
633  if (this->get<RiskOfInfectionFromSymptomatic>()[i] < 0.0 ||
634  this->get<RiskOfInfectionFromSymptomatic>()[i] > 1.0) {
635  log_warning("Constraint check: Parameter RiskOfInfectionFromSymptomatic smaller {:d} or larger {:d}", 0,
636  1);
637  }
638 
639  if (this->get<HospitalizedCasesPerInfectious>()[i] < 0.0 ||
640  this->get<HospitalizedCasesPerInfectious>()[i] > 1.0) {
641  log_warning("Constraint check: Parameter HospitalizedCasesPerInfectious smaller {:d} or larger {:d}", 0,
642  1);
643  }
644 
645  if (this->get<ICUCasesPerHospitalized>()[i] < 0.0 || this->get<ICUCasesPerHospitalized>()[i] > 1.0) {
646  log_warning("Constraint check: Parameter ICUCasesPerHospitalized smaller {:d} or larger {:d}", 0, 1);
647  }
648 
649  if (this->get<DeathsPerICU>()[i] < 0.0 || this->get<DeathsPerICU>()[i] > 1.0) {

652  }
