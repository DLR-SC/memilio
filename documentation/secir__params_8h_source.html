426  void apply_constraints()
427  {
428  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
429  log_warning("Constraint check: Parameter Seasonality changed from {:0.4f} to {:d}", this->get<Seasonality>(), 0);
430  this->set<Seasonality>(0);
431  }
432 
433  if (this->get<ICUCapacity>() < 0.0) {
434  log_warning("Constraint check: Parameter ICUCapacity changed from {:0.4f} to {:d}", this->get<ICUCapacity>(), 0);
435  this->set<ICUCapacity>(0);
436  }
437 
438  for(auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
439 
440  if (this->get<IncubationTime>()[i] < 2.0) {
441  log_warning("Constraint check: Parameter IncubationTime changed from {:.4f} to {:.4f}", this->get<IncubationTime>()[i], 2.0);
442  this->get<IncubationTime>()[i] = 2.0;
443  }
444 
445  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
446  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}", this->get<SerialInterval>()[i],
447  0.5 * this->get<IncubationTime>()[i] + 0.5);
448  this->get<SerialInterval>()[i] = 0.5 * this->get<IncubationTime>()[i] + 0.5;
449  }
450  else if (this->get<SerialInterval>()[i] > this->get<IncubationTime>()[i] - 0.5) {
451  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}", this->get<SerialInterval>()[i], this->get<IncubationTime>()[i] - 0.5);
452  this->get<SerialInterval>()[i] = this->get<IncubationTime>()[i] - 0.5;
453  }
454 
455  if (this->get<InfectiousTimeMild>()[i] < 1.0) {
456  log_warning("Constraint check: Parameter InfectiousTimeMild changed from {:.4f} to {:.4f}", this->get<InfectiousTimeMild>()[i], 1.0);
457  this->get<InfectiousTimeMild>()[i] = 1.0;
458  }
459 
460  if (this->get<HospitalizedToHomeTime>()[i] < 1.0) {

472  this->get<HospitalizedToICUTime>()[i] = 1.0;
473  }
474 
475  if (this->get<ICUToHomeTime>()[i] < 1.0) {
476  log_warning("Constraint check: Parameter ICUToHomeTime changed from {:.4f} to {:.4f}", this->get<ICUToHomeTime>()[i], 1.0);
477  this->get<ICUToHomeTime>()[i] = 1.0;
478  }
479

481  log_warning("Constraint check: Parameter ICUToDeathTime changed from {:.4f} to {:.4f}", this->get<ICUToDeathTime>()[i], 1.0);
482  this->get<ICUToDeathTime>()[i] = 1.0;
483  }

489  }
