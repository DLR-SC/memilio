426  void apply_constraints()
427  {
428  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
429  log_warning("Constraint check: Parameter Seasonality changed from {:0.4f} to {:d}", this->get<Seasonality>(), 0);
430  this->set<Seasonality>(0);
431  }
432 
433  if (this->get<ICUCapacity>() < 0.0) {
434  log_warning("Constraint check: Parameter ICUCapacity changed from {:0.4f} to {:d}", this->get<ICUCapacity>(), 0);
435  this->set<ICUCapacity>(0);
436  }
437 
438  for(auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
439 
440  if (this->get<IncubationTime>()[i] < 2.0) {
441  log_warning("Constraint check: Parameter IncubationTime changed from {:.4f} to {:.4f}", this->get<IncubationTime>()[i], 2.0);
442  this->get<IncubationTime>()[i] = 2.0;
443  }
444 
445  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
446  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}", this->get<SerialInterval>()[i],
447  0.5 * this->get<IncubationTime>()[i] + 0.5);
448  this->get<SerialInterval>()[i] = 0.5 * this->get<IncubationTime>()[i] + 0.5;
449  }
450  else if (this->get<SerialInterval>()[i] > this->get<IncubationTime>()[i] - 0.5) {
451  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}", this->get<SerialInterval>()[i], this->get<IncubationTime>()[i] - 0.5);
452  this->get<SerialInterval>()[i] = this->get<IncubationTime>()[i] - 0.5;
453  }
454 
455  if (this->get<InfectiousTimeMild>()[i] < 1.0) {
456  log_warning("Constraint check: Parameter InfectiousTimeMild changed from {:.4f} to {:.4f}", this->get<InfectiousTimeMild>()[i], 1.0);
457  this->get<InfectiousTimeMild>()[i] = 1.0;
458  }
459 
460  if (this->get<HospitalizedToHomeTime>()[i] < 1.0) {
461  log_warning("Constraint check: Parameter HospitalizedToHomeTime changed from {:.4f} to {:.4f}", this->get<HospitalizedToHomeTime>()[i], 1.0);
462  this->get<HospitalizedToHomeTime>()[i] = 1.0;
463  }
464

467  this->get<HomeToHospitalizedTime>()[i] = 1.0;
468  }
469

471  log_warning("Constraint check: Parameter HospitalizedToICUTime changed from {:.4f} to {:.4f}", this->get<HospitalizedToICUTime>()[i], 1.0);
472  this->get<HospitalizedToICUTime>()[i] = 1.0;
473  }
474 
475  if (this->get<ICUToHomeTime>()[i] < 1.0) {
476  log_warning("Constraint check: Parameter ICUToHomeTime changed from {:.4f} to {:.4f}", this->get<ICUToHomeTime>()[i], 1.0);
477  this->get<ICUToHomeTime>()[i] = 1.0;
478  }
479 
480  if (this->get<ICUToDeathTime>()[i] < 1.0) {
481  log_warning("Constraint check: Parameter ICUToDeathTime changed from {:.4f} to {:.4f}", this->get<ICUToDeathTime>()[i], 1.0);
482  this->get<ICUToDeathTime>()[i] = 1.0;
483  }
484 
485  auto t_inf_asymp = 1.0 / (0.5 / (this->get<IncubationTime>()[i] - this->get<SerialInterval>()[i])) + 0.5 * this->get<InfectiousTimeMild>()[i];
486  if (abs(this->get<InfectiousTimeAsymptomatic>()[i] - t_inf_asymp) > 1e-12) {

488  "https://www.medrxiv.org/content/10.1101/2020.04.04.20053637v1.");
489  this->get<InfectiousTimeAsymptomatic>()[i] = t_inf_asymp;
490  }
491 
492  if (this->get<InfectionProbabilityFromContact>()[i] < 0.0) {
493  log_warning("Constraint check: Parameter InfectionProbabilityFromContact changed from {:0.4f} to {:d} ", this->get<InfectionProbabilityFromContact>()[i], 0);
494  this->get<InfectionProbabilityFromContact>()[i] = 0;
495  }
496 
497  if (this->get<RelativeCarrierInfectability>()[i] < 0.0) {
498  log_warning("Constraint check: Parameter RelativeCarrierInfectability changed from {:0.4f} to {:d} ", this->get<RelativeCarrierInfectability>()[i], 0);
499  this->get<RelativeCarrierInfectability>()[i] = 0;
500  }

504  this->get<AsymptoticCasesPerInfectious>()[i] = 0;
505  }

508  log_warning("Constraint check: Parameter RiskOfInfectionFromSympomatic changed from {:0.4f} to {:d}", this->get<RiskOfInfectionFromSympomatic>()[i], 0);

532  void check_constraints() const
533  {
534  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
535  log_warning("Constraint check: Parameter m_seasonality smaller {:d} or larger {:d}", 0, 0.5);
536  }
537 
538  if (this->get<ICUCapacity>() < 0.0) {
539  log_warning("Constraint check: Parameter m_icu_capacity smaller {:d}", 0);
540  }
541 
542  for(auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
543 
544  if (this->get<IncubationTime>()[i] < 2.0) {
545  log_error("Constraint check: Parameter IncubationTime {:.4f} smaller {:.4f}", this->get<IncubationTime>()[i], 2.0);
546  }
547 
548  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
549  log_error("Constraint check: Parameter SerialInterval {:.4f} smaller {:.4f}", this->get<SerialInterval>()[i], 0.5 * this->get<IncubationTime>()[i] + 0.5);
550  }
551  else if (this->get<SerialInterval>()[i] > this->get<IncubationTime>()[i] - 0.5) {
552  log_error("Constraint check: Parameter SerialInterval {:.4f} smaller {:.4f}", this->get<SerialInterval>()[i], this->get<IncubationTime>()[i] - 0.5);
553  }
554 
555  if (this->get<InfectiousTimeMild>()[i] < 1.0) {
556  log_error("Constraint check: Parameter InfectiousTimeMild {:.4f} smaller {:.4f}", this->get<InfectiousTimeMild>()[i], 1.0);
557  }
558 
559  if (this->get<HospitalizedToHomeTime>()[i] < 1.0) {
560  log_error("Constraint check: Parameter HospitalizedToHomeTime {:.4f} smaller {:.4f}", this->get<HospitalizedToHomeTime>()[i], 1.0);
561  }
562 
563  if (this->get<HomeToHospitalizedTime>()[i] < 1.0) {
564  log_error("Constraint check: Parameter HomeToHospitalizedTime {:.4f} smaller {:.4f}", this->get<HomeToHospitalizedTime>()[i], 1.0);
565  }
566 
567  if (this->get<HospitalizedToICUTime>()[i] < 1.0) {

569  }
570 
571  if (this->get<ICUToHomeTime>()[i] < 1.0) {

574

576  log_warning("Constraint check: Parameter InfectiousTimeAsymptomatic not set as fully dependent on tinc, tserint and tinfmild, as proposed by "

578  }
579 
580 
581  if (this->get<ICUToDeathTime>()[i] < 1.0) {
582  log_error("Constraint check: Parameter ICUToDeathTime {:.4f} smaller {:.4f}", this->get<ICUToDeathTime>()[i], 1.0);
583  }
584 
585  if (this->get<InfectionProbabilityFromContact>()[i] < 0.0) {
586  log_warning("Constraint check: Parameter InfectionProbabilityFromContact smaller {:d}", 0);
587  }
588 
589  if (this->get<RelativeCarrierInfectability>()[i] < 0.0) {
590  log_warning("Constraint check: Parameter RelativeCarrierInfectability smaller {:d}", 0);
