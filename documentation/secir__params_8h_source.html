362  void apply_constraints()
363  {
364  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
365  log_warning("Constraint check: Parameter Seasonality changed from {:0.4f} to {:d}",
366  this->get<Seasonality>(), 0);
367  this->set<Seasonality>(0);
368  }
369 
370  if (this->get<ICUCapacity>() < 0.0) {
371  log_warning("Constraint check: Parameter ICUCapacity changed from {:0.4f} to {:d}",
372  this->get<ICUCapacity>(), 0);
373  this->set<ICUCapacity>(0);
374  }
375 
376  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
377 
378  if (this->get<IncubationTime>()[i] < 2.0) {
379  log_warning("Constraint check: Parameter IncubationTime changed from {:.4f} to {:.4f}",
380  this->get<IncubationTime>()[i], 2.0);
381  this->get<IncubationTime>()[i] = 2.0;
382  }
383 
384  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
385  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}",
386  this->get<SerialInterval>()[i], 0.5 * this->get<IncubationTime>()[i] + 0.5);
387  this->get<SerialInterval>()[i] = 0.5 * this->get<IncubationTime>()[i] + 0.5;
388  }

463  void check_constraints() const
464  {
465  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
466  log_warning("Constraint check: Parameter m_seasonality smaller {:d} or larger {:d}", 0, 0.5);
467  }
468 
469  if (this->get<ICUCapacity>() < 0.0) {
470  log_warning("Constraint check: Parameter m_icu_capacity smaller {:d}", 0);
471  }
472 
473  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
474 
475  if (this->get<IncubationTime>()[i] < 2.0) {
476  log_error("Constraint check: Parameter IncubationTime {:.4f} smaller {:.4f}",
477  this->get<IncubationTime>()[i], 2.0);
478  }
479 
480  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + 1.0) {
481  log_error("Constraint check: Parameter SerialInterval {:.4f} smaller {:.4f}",
482  this->get<SerialInterval>()[i], 0.5 * this->get<IncubationTime>()[i] + 0.5);
483  }
484  else if (this->get<SerialInterval>()[i] > this->get<IncubationTime>()[i] - 0.5) {
485  log_error("Constraint check: Parameter SerialInterval {:.4f} smaller {:.4f}",
486  this->get<SerialInterval>()[i], this->get<IncubationTime>()[i] - 0.5);
487  }
488 
489  if (this->get<TimeInfectedSymptoms>()[i] < 1.0) {
490  log_error("Constraint check: Parameter TimeInfectedSymptoms {:.4f} smaller {:.4f}",
491  this->get<TimeInfectedSymptoms>()[i], 1.0);
492  }
493 
494  if (this->get<TimeInfectedSevere>()[i] < 1.0) {
495  log_error("Constraint check: Parameter TimeInfectedSevere {:.4f} smaller {:.4f}",
496  this->get<TimeInfectedSevere>()[i], 1.0);
497  }
498 
499  if (this->get<TimeInfectedCritical>()[i] < 1.0) {
500  log_error("Constraint check: Parameter TimeInfectedCritical {:.4f} smaller {:.4f}",
501  this->get<TimeInfectedCritical>()[i], 1.0);
502  }
503 
504  if (this->get<TransmissionProbabilityOnContact>()[i] < 0.0) {
505  log_warning("Constraint check: Parameter TransmissionProbabilityOnContact smaller {:d}", 0);
506  }
507 
508  if (this->get<RelativeTransmissionNoSymptoms>()[i] < 0.0) {
509  log_warning("Constraint check: Parameter RelativeTransmissionNoSymptoms smaller {:d}", 0);
510  }
511 
512  if (this->get<RecoveredPerInfectedNoSymptoms>()[i] < 0.0 ||
513  this->get<RecoveredPerInfectedNoSymptoms>()[i] > 1.0) {
514  log_warning("Constraint check: Parameter RecoveredPerInfectedNoSymptoms smaller {:d} or larger {:d}", 0,
515  1);
516  }
517 
518  if (this->get<RiskOfInfectionFromSymptomatic>()[i] < 0.0 ||
519  this->get<RiskOfInfectionFromSymptomatic>()[i] > 1.0) {
520  log_warning("Constraint check: Parameter RiskOfInfectionFromSymptomatic smaller {:d} or larger {:d}", 0,
