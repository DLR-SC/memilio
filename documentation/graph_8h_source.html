157  template <class... Args>
158  Edge<EdgePropertyT>& add_edge(size_t start_node_idx, size_t end_node_idx, Args&&... args)
159  {
160  assert(m_nodes.size() > start_node_idx && m_nodes.size() > end_node_idx);
161  return *insert_sorted_replace(m_edges,
162  Edge<EdgePropertyT>(start_node_idx, end_node_idx, std::forward<Args>(args)...),
163  [](auto&& e1, auto&& e2) {
164  return e1.start_node_idx == e2.start_node_idx
165  ? e1.end_node_idx < e2.end_node_idx
166  : e1.start_node_idx < e2.start_node_idx;
167  });
168  }
