233  int check_constraints() const
234  {
235  if (this->get<TransmissionProbabilityOnContact>() < 0.0 ||
236  this->get<TransmissionProbabilityOnContact>() > 1.0) {
237  log_error("Constraint check: Parameter TransmissionProbabilityOnContact smaller {:d} or larger {:d}", 0, 1);
238  return 1;
239  }
240 
241  if (this->get<RelativeTransmissionNoSymptoms>() < 0.0 || this->get<RelativeTransmissionNoSymptoms>() > 1.0) {
242  log_error("Constraint check: Parameter RelativeTransmissionNoSymptoms smaller {:d} or larger {:d}", 0, 1);
243  return 1;
244  }
245 
246  if (this->get<RiskOfInfectionFromSymptomatic>() < 0.0 || this->get<RiskOfInfectionFromSymptomatic>() > 1.0) {
247  log_error("Constraint check: Parameter RiskOfInfectionFromSymptomatic smaller {:d} or larger {:d}", 0, 1);
248  return 1;
249  }
250 
251  for (int i = 0; i < (int)InfectionTransition::Count; i++) {
252  if (this->get<TransitionProbabilities>()[i] < 0.0 || this->get<TransitionProbabilities>()[i] > 1.0) {
253  log_error("Constraint check: One parameter TransitionProbabilities smaller {:d} or larger {:d}", 0, 1);
254  return 1;
255  }
256  }
257 
258  if (this->get<TransitionProbabilities>()[(int)InfectionTransition::SusceptibleToExposed] != 1.0) {
259  log_error("Constraint check: Parameter transitiion probability for SusceptibleToExposed unequal to {:d}",
260  1);
261  return 1;
262  }
263 
264  if (this->get<TransitionProbabilities>()[(int)InfectionTransition::ExposedToInfectedNoSymptoms] != 1.0) {
265  log_error(
266  "Constraint check: Parameter transitiion probability for ExposedToInfectedNoSymptoms unequal to {:d}",
267  1);
268  return 1;
269  }
270 
271  if (this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms] +
272  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedNoSymptomsToRecovered] !=
273  1.0) {
274  log_error("Constraint check: Sum of transitiion probability for InfectedNoSymptomsToInfectedSymptoms and "
275  "InfectedNoSymptomsToRecovered unequal to {:d}",
276  1);
277  return 1;
278  }
279 
280  if (this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSymptomsToInfectedSevere] +
281  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSymptomsToRecovered] !=
282  1.0) {
283  log_error("Constraint check: Sum of transitiion probability for InfectedSymptomsToInfectedSevere and "
284  "InfectedSymptomsToRecovered unequal to {:d}",
285  1);
286  return 1;
287  }
288 
289  if (this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSevereToInfectedCritical] +
290  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSevereToRecovered] !=
291  1.0) {
292  log_error("Constraint check: Sum of transitiion probability for InfectedSevereToInfectedCritical and "
293  "InfectedSevereToRecovered unequal to {:d}",
294  1);
295  return 1;
296  }
297 
298  if (this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedCriticalToDead] +
299  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedCriticalToRecovered] !=
300  1.0) {
301  log_error("Constraint check: Sum of transitiion probability for InfectedCriticalToDead and "
302  "InfectedCriticalToRecovered unequal to {:d}",
303  1);
304  return 1;
305  }
306 
307  return 0;
