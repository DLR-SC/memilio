222  double(d.get_time()) < double(end)) {

240 template <class DampingExpr>
241 Eigen::Ref<const typename DampingExpr::Matrix> get_active_damping(const DampingExpr& damping_expr, DampingLevel lvl,
242  DampingType type, SimulationTime t)
243 {
244  auto ub =
245  std::find_if(damping_expr.get_dampings().rbegin(), damping_expr.get_dampings().rend(), [lvl, type, t](auto& d) {
246  return d.get_level() == lvl && d.get_type() == type && double(d.get_time()) <= double(t);
247  });
248  if (ub != damping_expr.get_dampings().rend()) {
249  return ub->get_coeffs();
250  }
251  return DampingExpr::Matrix::Zero(damping_expr.get_shape().rows(), damping_expr.get_shape().cols());
252 }

280 template <class DampingExprGroup, class MakeMatrix>
281 void implement_dynamic_npis(DampingExprGroup& damping_expr_group, const std::vector<DampingSampling>& npis,
282  SimulationTime begin, SimulationTime end, MakeMatrix&& make_matrix)
283 {
284  for (auto& npi : npis) {
285  for (auto& mat_idx : npi.get_matrix_indices()) {
286  auto type = npi.get_type();
287  auto level = npi.get_level();
288  auto& damping_expr = damping_expr_group[mat_idx];
289 
290  auto active = get_active_damping(damping_expr, level, type, begin);
291  auto active_end =
292  get_active_damping(damping_expr, level, type, end).eval(); //copy because it may be removed or changed
293  auto value = make_matrix(double(npi.get_value()) * npi.get_group_weights());
294 
295  auto npi_implemented = false;
296 
297  //add begin of npi if not already bigger
298  if ((active.array() < value.array()).any()) {
299  damping_expr.add_damping(max(value, active), level, type, begin);
300  npi_implemented = true;
301  }
302 
303  //replace dampings during the new npi
304  auto damping_indices = get_damping_indices(damping_expr, level, type, begin, end);
305  for (auto& i : damping_indices) {
306  auto& d = damping_expr.get_dampings()[i];
307  damping_expr.add_damping(max(d.get_coeffs(), value), level, type, d.get_time());
308  npi_implemented = true;
309  }
310 
311  //add end of npi to restore active dampings if any change was made
312  if (npi_implemented) {
313  damping_expr.add_damping(active_end, level, type, end);
314  }
315  }
316  }
317 
318  //remove duplicates that accumulated because of dampings that become active during the time span
319  //a damping is obsolete if the most recent damping of the same type and level has the same value
320  for (auto& damping_expr : damping_expr_group) {
321  //go from the back so indices aren't invalidated when dampings are removed
322  //use indices to loop instead of reverse iterators because removing invalidates the current iterator
323  for (auto i = size_t(0); i < damping_expr.get_dampings().size() - 1; ++i) {
324  auto it = damping_expr.get_dampings().rbegin() + i;
325 
326  //look for previous damping of the same type/level
327  auto it_prev = std::find_if(it + 1, damping_expr.get_dampings().rend(), [&di = *it](auto& dj) {
328  return di.get_level() == dj.get_level() && di.get_type() == dj.get_type();
329  });
330 
331  //remove if match is found and has same value
332  if (it_prev != damping_expr.get_dampings().rend() && it->get_coeffs() == it_prev->get_coeffs()) {
333  damping_expr.remove_damping(damping_expr.get_dampings().size() - 1 - i);
334  }
335  }
336  }
337 }
