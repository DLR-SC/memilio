227  auto obj = io.expect_object("DampingMatrixExpression");
228  auto b = obj.expect_element("Baseline", Tag<Matrix>{});
229  auto m = obj.expect_element("Minimum", Tag<Matrix>{});
230  auto d = obj.expect_list("Dampings", Tag<typename DampingsType::value_type>{});
231  return apply(io, [](auto&& b_, auto&& m_, auto&& d_) -> IOResult<Derived> {
232  if (Shape::get_shape_of(b_) != Shape::get_shape_of(m_)) {
233  return failure(StatusCode::InvalidValue, "Baseline and Minimum must have the same shape.");
234  }
235  auto r = Derived(b_, m_);
236  for (auto&& e : d_) {
237  if (e.get_shape() != Shape::get_shape_of(b_)) {
238  return failure(StatusCode::InvalidValue, "Dampings must have the same shape as the Baseline.");
239  }
240  r.add_damping(e);
241  }
242  return success(r);
243  }, b, m, d);
