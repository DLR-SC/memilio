218  /*
219  * @brief verification that at least 99% of the density
220  * function lie in the interval defined by the boundaries
221  */
222  bool check_quantiles(double& mean, double& standard_dev)
223  {
224  bool changed = false;
225 
226  changed = adapt_mean(mean);
227 
228  changed = adapt_standard_dev(standard_dev);
229 
230  if (changed && m_log_stddev_change) {
231  log_warning("Standard deviation reduced to fit 99% of the distribution within [lowerbound,upperbound].");
232  }
233 
234  return changed;
235  }
236 
237  bool adapt_mean(double& mean)
238  {
239  bool changed = false;
240  if (mean < m_lower_bound || mean > m_upper_bound) {
241  mean = 0.5 * (m_upper_bound - m_lower_bound);
242  changed = true;
243  }
244  return changed;
245  }
246 
247  // ensure that 0.99 % of the distribution are within lower bound and upper bound
248  bool adapt_standard_dev(double& standard_dev)
249  {
250  bool changed = false;
251  if (m_mean + standard_dev * m_quantile > m_upper_bound) {
252  standard_dev = (m_upper_bound - m_mean) / m_quantile;
253  changed = true;
254  }
255  if (m_mean - standard_dev * m_quantile < m_lower_bound) {
256  standard_dev = (m_mean - m_lower_bound) / m_quantile;
257  changed = true;
258  }
259 
260  return changed;
261  }
262 
263  void set_standard_dev(double standard_dev)
264  {
265  m_standard_dev = standard_dev;
266  }
267

269  {
270  return m_mean;
271  }

275  return m_standard_dev;

283  /*
284  * @brief gets a sample of a normally distributed variable
285  * before sampling, it is verified that at least 99% of the
286  * density function lie in the interval defined by the boundaries
287  * otherwise the normal distribution is adapted
288  */
289  double get_rand_sample() override
290  {
291  if (check_quantiles(m_mean, m_standard_dev) || m_distribution.mean() != m_mean ||
292  m_distribution.stddev() != m_standard_dev) {
293  m_distribution = std::normal_distribution<double>{m_mean, m_standard_dev};
294  }
295 
296  int i = 0;
297  int retries = 10;
298  double rnumb = m_distribution(thread_local_rng());
299  while ((rnumb > m_upper_bound || rnumb < m_lower_bound) && i < retries) {
300  rnumb = m_distribution(thread_local_rng());
301  i++;
302  if (i == retries) {
303  log_warning("Not successfully sampled within [min,max].");
304  if (rnumb > m_upper_bound) {
305  rnumb = m_upper_bound;
306  }
307  else {
308  rnumb = m_lower_bound;
309  }
310  }
311  }
312  return rnumb;
313  }
314 
315  ParameterDistribution* clone() const override
316  {
317  return new ParameterDistributionNormal(*this);
318  }
319 
320 private:
321  double m_mean; // the mean value of the normal distribution
322  double m_standard_dev; // the standard deviation of the normal distribution
323  constexpr static double m_quantile = 2.5758; // 0.995 quartile
324  std::normal_distribution<double> m_distribution;
325  bool m_log_stddev_change = true;
326 };

384 template <class IOContext>
385 IOResult<std::shared_ptr<ParameterDistribution>> deserialize_internal(IOContext& io,
386  Tag<std::shared_ptr<ParameterDistribution>>)
387 {
388  auto obj = io.expect_object("ParameterDistribution");
389  auto type = obj.expect_element("Type", Tag<std::string>{});
390  if (type) {
391  if (type.value() == "Uniform") {
392  auto lb = obj.expect_element("LowerBound", Tag<double>{});
393  auto ub = obj.expect_element("UpperBound", Tag<double>{});
394  auto predef = obj.expect_list("PredefinedSamples", Tag<double>{});
395  auto p = apply(
396  io,
397  [](auto&& lb_, auto&& ub_, auto&& predef_) {
398  auto distr = std::make_shared<ParameterDistributionUniform>(lb_, ub_);
399  for (auto&& e : predef_) {
400  distr->add_predefined_sample(e);
401  }
402  return distr;
403  },
404  lb, ub, predef);
405  if (p) {
406  return success(p.value());
407  }
408  else {
409  return p.as_failure();
410  }
411  }
412  else if (type.value() == "Normal") {
413  auto m = obj.expect_element("Mean", Tag<double>{});
414  auto s = obj.expect_element("StandardDev", Tag<double>{});
415  auto lb = obj.expect_element("LowerBound", Tag<double>{});
416  auto ub = obj.expect_element("UpperBound", Tag<double>{});
417  auto predef = obj.expect_list("PredefinedSamples", Tag<double>{});
418  auto p = apply(
419  io,
420  [](auto&& lb_, auto&& ub_, auto&& m_, auto&& s_, auto&& predef_) {
421  auto distr = std::make_shared<ParameterDistributionNormal>(lb_, ub_, m_, s_);
422  for (auto&& e : predef_) {
423  distr->add_predefined_sample(e);
424  }
425  return distr;
426  },
427  lb, ub, m, s, predef);
428  if (p) {
429  return success(p.value());
430  }
431  else {
432  return p.as_failure();
433  }
434  }
435  else {
436  return failure(StatusCode::InvalidValue, "Type of ParameterDistribution " + type.value() + " not valid.");
437  }
438  }
439  return failure(type.error());
440 }
