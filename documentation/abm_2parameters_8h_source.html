633  bool check_constraints() const
634  {
635  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
636 
637  if (this->get<IncubationPeriod>()[{VirusVariant::Wildtype, i}] < 0) {
638  log_error("Constraint check: Parameter IncubationPeriod of age group {:.0f} smaller than {:.4f}",
639  (size_t)i, 0);
640  return true;
641  }
642 
643  if (this->get<InfectedNoSymptomsToSymptoms>()[{VirusVariant::Wildtype, i}] < 0.0) {
644  log_error("Constraint check: Parameter InfectedNoSymptomsToSymptoms of age group {:.0f} smaller "
645  "than {:d}",
646  (size_t)i, 0);
647  return true;
648  }
649 
650  if (this->get<InfectedNoSymptomsToRecovered>()[{VirusVariant::Wildtype, i}] < 0.0) {
651  log_error("Constraint check: Parameter InfectedNoSymptomsToRecovered of age group {:.0f} smaller "
652  "than {:d}",
653  (size_t)i, 0);
654  return true;
655  }
656 
657  if (this->get<InfectedSymptomsToRecovered>()[{VirusVariant::Wildtype, i}] < 0.0) {
658  log_error(
659  "Constraint check: Parameter InfectedSymptomsToRecovered of age group {:.0f} smaller than {:d}",
660  (size_t)i, 0);
661  return true;
662  }
663 
664  if (this->get<InfectedSymptomsToSevere>()[{VirusVariant::Wildtype, i}] < 0.0) {
665  log_error("Constraint check: Parameter InfectedSymptomsToSevere of age group {:.0f} smaller than {:d}",
666  (size_t)i, 0);
667  return true;
668  }
669 
670  if (this->get<SevereToCritical>()[{VirusVariant::Wildtype, i}] < 0.0) {
671  log_error("Constraint check: Parameter SevereToCritical of age group {:.0f} smaller than {:d}",
672  (size_t)i, 0);
673  return true;
674  }
675 
676  if (this->get<SevereToRecovered>()[{VirusVariant::Wildtype, i}] < 0.0) {
677  log_error("Constraint check: Parameter SevereToRecovered of age group {:.0f} smaller than {:d}",
678  (size_t)i, 0);
679  return true;
680  }
681 
682  if (this->get<CriticalToDead>()[{VirusVariant::Wildtype, i}] < 0.0) {
683  log_error("Constraint check: Parameter CriticalToDead of age group {:.0f} smaller than {:d}", (size_t)i,
684  0);
685  return true;
686  }
687 
688  if (this->get<CriticalToRecovered>()[{VirusVariant::Wildtype, i}] < 0.0) {
689  log_error("Constraint check: Parameter CriticalToRecovered of age group {:.0f} smaller than {:d}",
690  (size_t)i, 0);
691  return true;
692  }
693 
694  if (this->get<RecoveredToSusceptible>()[{VirusVariant::Wildtype, i}] < 0.0) {
695  log_error("Constraint check: Parameter RecoveredToSusceptible of age group {:.0f} smaller than {:d}",
696  (size_t)i, 0);
697  return true;
698  }
699 
700  if (this->get<DetectInfection>()[{VirusVariant::Wildtype, i}] < 0.0 ||
701  this->get<DetectInfection>()[{VirusVariant::Wildtype, i}] > 1.0) {
702  log_error("Constraint check: Parameter DetectInfection of age group {:.0f} smaller than {:d} or "
703  "larger than {:d}",
704  (size_t)i, 0, 1);
705  return true;
706  }
707 
708  if (this->get<GotoWorkTimeMinimum>()[i].seconds() < 0.0 ||
709  this->get<GotoWorkTimeMinimum>()[i].seconds() > this->get<GotoWorkTimeMaximum>()[i].seconds()) {
710  log_error("Constraint check: Parameter GotoWorkTimeMinimum of age group {:.0f} smaller {:d} or "

720  (size_t)i, this->get<GotoWorkTimeMinimum>()[i].seconds());
721  return true;
722  }
723 
724  if (this->get<GotoSchoolTimeMinimum>()[i].seconds() < 0.0 ||
725  this->get<GotoSchoolTimeMinimum>()[i].seconds() > this->get<GotoSchoolTimeMaximum>()[i].seconds()) {
726  log_error("Constraint check: Parameter GotoSchoolTimeMinimum of age group {:.0f} smaller {:d} or "

729  return true;
730  }
731

733  this->get<GotoSchoolTimeMaximum>()[i] > days(1)) {
734  log_error("Constraint check: Parameter GotoWorkTimeMaximum of age group {:.0f} smaller {:d} or larger "

742  this->get<MaskProtection>()[MaskType::Community] > 1.0) {

754 
755  if (this->get<MaskProtection>()[MaskType::Surgical] < 0.0 ||

762  if (this->get<LockdownDate>().seconds() < 0.0) {
763  log_error("Constraint check: Parameter LockdownDate smaller {:d}", 0);
764  return true;
765  }
766 
767  return false;
768  }
